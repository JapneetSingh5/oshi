49      /**
50       * Test operating system
51       */
52      @Test
53      public void testOperatingSystem() {
54          SystemInfo si = new SystemInfo();
55          OperatingSystem os = si.getOperatingSystem();
56          assertNotNull(os.getFamily());
57          assertNotNull(os.getManufacturer());
58          OSVersionInfo versionInfo = os.getVersionInfo();
59          assertNotNull(versionInfo);
60  
61          assertTrue(os.getSystemUptime() > 0);
62          assertTrue(os.getSystemBootTime() > 0);
63          assertTrue(os.getSystemBootTime() < System.currentTimeMillis() / 1000L);
64  
65          assertTrue(os.getProcessCount() >= 1);
66          assertTrue(os.getThreadCount() >= 1);
67          assertTrue(os.getBitness() == 32 || os.getBitness() == 64);
68          assertTrue(os.getProcessId() > 0);
69          assertEquals(os.isElevated(), os.isElevated());
70  
71          assertFalse(os.getProcesses(0, null).isEmpty());
72          OSProcess proc = os.getProcess(os.getProcessId());
73          assertTrue(proc.getName().length() > 0);
74          assertTrue(proc.getPath().length() > 0);
75          assertNotNull(proc.getCommandLine());
76          assertNotNull(proc.getCurrentWorkingDirectory());
77          assertNotNull(proc.getUser());
78          assertNotNull(proc.getUserID());
79          assertNotNull(proc.getGroup());
80          assertNotNull(proc.getGroupID());
81          assertNotEquals(State.INVALID, proc.getState());
82          assertEquals(proc.getProcessID(), os.getProcessId());
83          assertTrue(proc.getParentProcessID() >= 0);
84          assertTrue(proc.getThreadCount() > 0);
85          assertTrue(proc.getPriority() >= -20 && proc.getPriority() <= 128);
86          assertTrue(proc.getVirtualSize() >= 0);
87          assertTrue(proc.getResidentSetSize() >= 0);
88          assertTrue(proc.getKernelTime() >= 0);
89          assertTrue(proc.getUserTime() >= 0);
90          assertTrue(proc.getUpTime() >= 0);
91          assertTrue(proc.getProcessCpuLoadCumulative() >= 0d);
92          assertEquals(proc.getProcessCpuLoadCumulative(), proc.getProcessCpuLoadBetweenTicks(null), Double.MIN_VALUE);
93          assertEquals(proc.getProcessCpuLoadCumulative(), proc.getProcessCpuLoadBetweenTicks(proc), Double.MIN_VALUE);
94          assertTrue(proc.getStartTime() >= 0);
95          assertTrue(proc.getBytesRead() >= 0);
96          assertTrue(proc.getBytesWritten() >= 0);
97          assertTrue("Process bitness can't exceed OS bitness", proc.getBitness() <= os.getBitness());
98          assertEquals("Bitness must be 0, 32 or 64", 0, proc.getBitness() & ~(32 | 64));
99          assertTrue(proc.getOpenFiles() >= -1);
100     }

102     /**
103      * Tests process query by pid list
104      */
105     @Test
106     public void testProcessQueryByList() {
107         SystemInfo si = new SystemInfo();
108         OperatingSystem os = si.getOperatingSystem();
109         assertNotNull(os.getFamily());
110         assertNotNull(os.getManufacturer());
111         OSVersionInfo versionInfo = os.getVersionInfo();
112         assertNotNull(versionInfo);
113 
114         assertTrue(os.getProcessCount() >= 1);
115         assertTrue(os.getThreadCount() >= 1);
116         assertTrue(os.getProcessId() > 0);
117 
118         List<OSProcess> processes = os.getProcesses(5, null);
119         assertNotNull(processes);
120         // every OS should have at least one process running on it
121         assertFalse(processes.isEmpty());
122         // the list of pids we want info on
123         List<Integer> pids = new ArrayList<>();
124         for (OSProcess p : processes) {
125             pids.add(p.getProcessID());
126         }
127         // query for just those processes
128         Collection<OSProcess> processes1 = os.getProcesses(pids);
129         // there's a potential for a race condition here, if a process we
130         // queried for initially wasn't running during the second query. In this case,
131         // try again with the shorter list
132         while (processes1.size() < pids.size()) {
133             pids.clear();
134             for (OSProcess p : processes1) {
135                 pids.add(p.getProcessID());
136             }
137             // query for just those processes
138             processes1 = os.getProcesses(pids);
139         }
140         assertEquals(processes1.size(), pids.size());
141 
142     }

144     /**
145      * Tests child process getter
146      */
147     @Test
148     public void testGetChildProcesses() {
149         // Testing child processes is tricky because we don't really know a priori what
150         // processes might have children, and if we do test the full list vs. individual
151         // processes, we run into a race condition where child processes can start or
152         // stop before we measure a second time. So we can't really test for one-to-one
153         // correspondence of child process lists.
154         //
155         // We can expect code logic failures to occur all/most of the time for
156         // categories of processes, however, and allow occasional differences due to
157         // race conditions. So we will test three categories of processes: Those with 0
158         // children, those with exactly 1 child process, and those with multiple child
159         // processes. On the second poll, we expect at least half of processes in those
160         // categories to still be in the same category.
161         //
162         SystemInfo si = new SystemInfo();
163         OperatingSystem os = si.getOperatingSystem();
164         List<OSProcess> processes = os.getProcesses(0, null);
165         Set<Integer> zeroChildSet = new HashSet<>();
166         Set<Integer> oneChildSet = new HashSet<>();
167         Set<Integer> manyChildSet = new HashSet<>();
168         // Initialize all processes with no children
169         for (OSProcess p : processes) {
170             zeroChildSet.add(p.getProcessID());
171         }
172         // Move parents with 1 or more children to other set
173         for (OSProcess p : processes) {
174             if (zeroChildSet.contains(p.getParentProcessID())) {
175                 // Zero to One
176                 zeroChildSet.remove(p.getParentProcessID());
177                 oneChildSet.add(p.getParentProcessID());
178             } else if (oneChildSet.contains(p.getParentProcessID())) {
179                 // One to many
180                 oneChildSet.remove(p.getParentProcessID());
181                 manyChildSet.add(p.getParentProcessID());
182             }
183         }
184         // Now test that majority of each set is in same category
185         int matched = 0;
186         int total = 0;
187         for (Integer i : zeroChildSet) {
188             if (os.getChildProcesses(i, 0, null).isEmpty()) {
189                 matched++;
190             }
191             // Quit if enough to test
192             if (++total > 9) {
193                 break;
194             }
195         }
196         if (total > 4) {
197             assertTrue("Most processes with no children should not suddenly have them.", matched > total / 2);
198         }
199         matched = 0;
200         total = 0;
201         for (Integer i : oneChildSet) {
202             if (os.getChildProcesses(i, 0, null).size() == 1) {
203                 matched++;
204             }
205             // Quit if enough to test
206             if (++total > 9) {
207                 break;
208             }
209         }
210         if (total > 4) {
211             assertTrue("Most processes with one child should not suddenly have zero or more than one.",
212                     matched > total / 2);
213         }
214         matched = 0;
215         total = 0;
216         for (Integer i : manyChildSet) {
217             if (os.getChildProcesses(i, 0, null).size() > 1) {
218                 matched++;
219             }
220             // Quit if enough to test
221             if (++total > 9) {
222                 break;
223             }
224         }
225         if (total > 4) {
226             assertTrue("Most processes with more than one child should not suddenly have one or less.",
227                     matched > total / 2);
228         }
229     }

231     @Test
232     public void testGetCommandLine() {
233         int processesWithNonEmptyCmdLine = 0;
234 
235         SystemInfo si = new SystemInfo();
236         OperatingSystem os = si.getOperatingSystem();
237         for (OSProcess process : os.getProcesses(0, null)) {
238             if (!process.getCommandLine().trim().isEmpty()) {
239                 processesWithNonEmptyCmdLine++;
240             }
241         }
242 
243         assertTrue(processesWithNonEmptyCmdLine >= 1);
244     }

246     /**
247      * Tests services getter
248      */
249     @Test
250     public void testGetServices() {
251         SystemInfo si = new SystemInfo();
252         OperatingSystem os = si.getOperatingSystem();
253         int stopped = 0;
254         int running = 0;
255         for (OSService svc : os.getServices()) {
256             assertTrue(svc.getName().length() > 0);
257             switch (svc.getState()) {
258             case STOPPED:
259                 stopped++;
260                 break;
261             case RUNNING:
262                 running++;
263                 break;
264             default:
265                 break;
266             }
267         }
268         // Should be at least one of each
269         assertNotEquals(0, stopped);
270         assertNotEquals(0, running);
271     }
