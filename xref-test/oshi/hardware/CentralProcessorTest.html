24  package oshi.hardware;
25  
26  import static org.junit.Assert.assertEquals;
27  import static org.junit.Assert.assertFalse;
28  import static org.junit.Assert.assertNotNull;
29  import static org.junit.Assert.assertTrue;
30  
31  import org.junit.Test;
32  
33  import oshi.SystemInfo;
34  import oshi.hardware.CentralProcessor.ProcessorIdentifier;
35  import oshi.hardware.CentralProcessor.TickType;
36  import oshi.util.Util;
37  
38  /**
39   * Test CPU
40   */
41  public class CentralProcessorTest {
42  
43      /**
44       * Test central processor.
45       */
46      @Test
47      public void testCentralProcessor() {
48          SystemInfo si = new SystemInfo();
49          CentralProcessor p = si.getHardware().getProcessor();
50  
51          ProcessorIdentifier pi = p.getProcessorIdentifier();
52          assertNotNull("Processor Identifier's vendor shouldn't be Null", pi.getVendor());
53          assertTrue("Processor Identifier's vendor frequency should either be -1, or a greater than 0",
54                  pi.getVendorFreq() == -1 || pi.getVendorFreq() > 0);
55          assertNotNull("Processor Identifier's name shouldn't be Null", pi.getName());
56          assertNotNull("Processor Identifier's identifier shouldn't be Null", pi.getIdentifier());
57          assertNotNull("Processor Identifier's ID shouldn't be Null", pi.getProcessorID());
58          assertNotNull("Processor Identifier's stepping shouldn't be Null", pi.getStepping());
59          assertNotNull("Processor Identifier's model shouldn't be null", pi.getModel());
60          assertNotNull("Processor Identifier's family shouldn't be null", pi.getFamily());
61          assertNotNull("Processor Identifier's toString shouldn't be null", pi.toString());
62          assertFalse("Processor Identifier's micro-architecture shouldn't be blank",
63                  Util.isBlank(pi.getMicroarchitecture()));
64  
65          long[] ticks = p.getSystemCpuLoadTicks();
66          long[][] procTicks = p.getProcessorCpuLoadTicks();
67          assertEquals("System should have the same amount of cpu-load-tick counters as there are TickType values",
68                  ticks.length, TickType.values().length);
69  
70          Util.sleep(500);
71  
72          assertTrue("System's cpu load between ticks should be inclusively between 0 and 1",
73                  p.getSystemCpuLoadBetweenTicks(ticks) >= 0 && p.getSystemCpuLoadBetweenTicks(ticks) <= 1);
74          assertEquals("System's load averages length for 3 elements should equal 3", 3,
75                  p.getSystemLoadAverage(3).length);
76  
77          assertEquals("Central Processor's cpu load between ticks should equal the logical processor count",
78                  p.getProcessorCpuLoadBetweenTicks(procTicks).length, p.getLogicalProcessorCount());
79          for (int cpu = 0; cpu < p.getLogicalProcessorCount(); cpu++) {
80              assertTrue(
81                      "Central Processor's cpu number " + cpu
82                              + "'s load between ticks should be inclusively between 0 and 1",
83                      p.getProcessorCpuLoadBetweenTicks(procTicks)[cpu] >= 0
84                              && p.getProcessorCpuLoadBetweenTicks(procTicks)[cpu] <= 1);
85              assertEquals(
86                      "Central Processor's cpu number " + cpu
87                              + " should have the same amount of cpu-load-tick counters as there are TickType values",
88                      p.getProcessorCpuLoadTicks()[cpu].length, TickType.values().length);
89          }
90  
91          assertTrue(
92                  "Central Processor's logical processor count should be at least as high as its physical processor count",
93                  p.getLogicalProcessorCount() >= p.getPhysicalProcessorCount());
94          assertTrue("Central Processor's physical processor count should by higher than 0",
95                  p.getPhysicalProcessorCount() > 0);
96          assertTrue("Central Processor's physical processor count should be higher than its physical package count",
97                  p.getPhysicalProcessorCount() >= p.getPhysicalPackageCount());
98          assertTrue("Central Processor's physical package count should be higher than 0",
99                  p.getPhysicalPackageCount() > 0);
100         assertTrue("Central Processor's context switch count should be 0 or higher", p.getContextSwitches() >= 0);
101         assertTrue("Central Processor's interrupt count should be 0 or higher", p.getInterrupts() >= 0);
102 
103         long max = p.getMaxFreq();
104         long[] curr = p.getCurrentFreq();
105         assertEquals(
106                 "Central Processor's logical processor frequency array length should be the same as its logical processor count",
107                 curr.length, p.getLogicalProcessorCount());
108         if (max >= 0) {
109             for (int i = 0; i < curr.length; i++) {
110                 assertTrue("Central Processor's logical processor frequency should be at most it's max frequency",
111                         curr[i] <= max);
112             }
113         }
114 
115         for (int lp = 0; lp < p.getLogicalProcessorCount(); lp++) {
116             assertTrue("Logical processor number is negative", p.getLogicalProcessors()[lp].getProcessorNumber() >= 0);
117             switch (SystemInfo.getCurrentPlatformEnum()) {
118             case WINDOWS:
119                 if (p.getLogicalProcessorCount() < 64) {
120                     assertEquals("Processor group should be 0 for Windows systems with less than 64 logical processors",
121                             0, p.getLogicalProcessors()[lp].getProcessorGroup());
122                 }
123                 assertTrue("NUMA node number is negative", p.getLogicalProcessors()[lp].getNumaNode() >= 0);
124                 break;
125             case LINUX:
126                 assertEquals("Processor group should be 0 for Linux systems", 0,
127                         p.getLogicalProcessors()[lp].getProcessorGroup());
128                 assertTrue("NUMA node number is negative", p.getLogicalProcessors()[lp].getNumaNode() >= 0);
129                 break;
130             case MACOSX:
131                 assertEquals("Processor group should be 0 for macOS systems", 0,
132                         p.getLogicalProcessors()[lp].getProcessorGroup());
133                 assertEquals("NUMA Node should be 0 for macOS systems", 0, p.getLogicalProcessors()[lp].getNumaNode());
134                 break;
135             case SOLARIS:
136                 assertEquals("Processor group should be 0 for Solaris systems", 0,
137                         p.getLogicalProcessors()[lp].getProcessorGroup());
138                 assertTrue("NUMA node number is negative", p.getLogicalProcessors()[lp].getNumaNode() >= 0);
139                 break;
140             case FREEBSD:
141                 assertEquals("Processor group should be 0 for FreeBSD systems", 0,
142                         p.getLogicalProcessors()[lp].getProcessorGroup());
143                 assertEquals("NUMA Node should be 0 for FreeBSD systems", 0,
144                         p.getLogicalProcessors()[lp].getNumaNode());
145                 break;
146             default:
147                 break;
148             }
149         }
150     }
151 }
