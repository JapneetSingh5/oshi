40  /**
41   * Utility to read process data from HKEY_PERFORMANCE_DATA information with
42   * backup from Performance Counters or WMI
43   */
44  @ThreadSafe
45  public final class ProcessPerformanceData {
46  
47      private static final String PROCESS = "Process";
48  
49      private ProcessPerformanceData() {
50      }
51  
52      /**
53       * Query the registry for process performance counters
54       *
55       * @param pids
56       *            An optional collection of process IDs to filter the list to. May
57       *            be null for no filtering.
58       * @return A map with Process ID as the key and a {@link PerfCounterBlock}
59       *         object populated with performance counter information if successful,
60       *         or null otherwise.
61       */
62      public static Map<Integer, PerfCounterBlock> buildProcessMapFromRegistry(Collection<Integer> pids) {
63          // Grab the data from the registry.
64          Triplet<List<Map<ProcessPerformanceProperty, Object>>, Long, Long> processData = HkeyPerformanceDataUtil
65                  .readPerfDataFromRegistry(PROCESS, ProcessPerformanceProperty.class);
66          if (processData == null) {
67              return null;
68          }
69          List<Map<ProcessPerformanceProperty, Object>> processInstanceMaps = processData.getA();
70          long perfTime100nSec = processData.getB(); // 1601
71          long now = processData.getC(); // 1970 epoch
72  
73          // Create a map and fill it
74          Map<Integer, PerfCounterBlock> processMap = new HashMap<>();
75          // Iterate instances.
76          for (Map<ProcessPerformanceProperty, Object> processInstanceMap : processInstanceMaps) {
77              int pid = ((Integer) processInstanceMap.get(ProcessPerformanceProperty.PROCESSID)).intValue();
78              String name = (String) processInstanceMap.get(ProcessPerformanceProperty.NAME);
79              if ((pids == null || pids.contains(pid)) && !"_Total".equals(name)) {
80                  long upTime = (perfTime100nSec - (Long) processInstanceMap.get(ProcessPerformanceProperty.CREATIONDATE))
81                          / 10_000L;
82                  if (upTime < 1L) {
83                      upTime = 1L;
84                  }
85                  processMap.put(pid, new PerfCounterBlock(name,
86                          (Integer) processInstanceMap.get(ProcessPerformanceProperty.PARENTPROCESSID),
87                          (Integer) processInstanceMap.get(ProcessPerformanceProperty.PRIORITY),
88                          (Long) processInstanceMap.get(ProcessPerformanceProperty.PRIVATEPAGECOUNT), now - upTime,
89                          upTime, (Long) processInstanceMap.get(ProcessPerformanceProperty.READTRANSFERCOUNT),
90                          (Long) processInstanceMap.get(ProcessPerformanceProperty.WRITETRANSFERCOUNT)));
91              }
92          }
93          return processMap;
94      }
95  
96      /**
97       * Query PerfMon for process performance counters
98       *
99       * @param pids
100      *            An optional collection of process IDs to filter the list to. May
101      *            be null for no filtering.
102      * @return A map with Process ID as the key and a {@link PerfCounterBlock}
103      *         object populated with performance counter information.
104      */
105     public static Map<Integer, PerfCounterBlock> buildProcessMapFromPerfCounters(Collection<Integer> pids) {
106         Map<Integer, PerfCounterBlock> processMap = new HashMap<>();
107         Pair<List<String>, Map<ProcessPerformanceProperty, List<Long>>> instanceValues = ProcessInformation
108                 .queryProcessCounters();
109         long now = System.currentTimeMillis(); // 1970 epoch
110         List<String> instances = instanceValues.getA();
111         Map<ProcessPerformanceProperty, List<Long>> valueMap = instanceValues.getB();
112         List<Long> pidList = valueMap.get(ProcessPerformanceProperty.PROCESSID);
113         List<Long> ppidList = valueMap.get(ProcessPerformanceProperty.PARENTPROCESSID);
114         List<Long> priorityList = valueMap.get(ProcessPerformanceProperty.PRIORITY);
115         List<Long> ioReadList = valueMap.get(ProcessPerformanceProperty.READTRANSFERCOUNT);
116         List<Long> ioWriteList = valueMap.get(ProcessPerformanceProperty.WRITETRANSFERCOUNT);
117         List<Long> workingSetSizeList = valueMap.get(ProcessPerformanceProperty.PRIVATEPAGECOUNT);
118         List<Long> creationTimeList = valueMap.get(ProcessPerformanceProperty.CREATIONDATE);
119 
120         for (int inst = 0; inst < instances.size(); inst++) {
121             int pid = pidList.get(inst).intValue();
122             if (pids == null || pids.contains(pid)) {
123                 // if creation time value is less than current millis, it's in 1970 epoch,
124                 // otherwise it's 1601 epoch and we must convert
125                 long ctime = creationTimeList.get(inst);
126                 if (ctime > now) {
127                     ctime = WinBase.FILETIME.filetimeToDate((int) (ctime >> 32), (int) (ctime & 0xffffffffL)).getTime();
128                 }
129                 processMap.put(pid,
130                         new PerfCounterBlock(instances.get(inst), ppidList.get(inst).intValue(),
131                                 priorityList.get(inst).intValue(), workingSetSizeList.get(inst), ctime, now - ctime,
132                                 ioReadList.get(inst), ioWriteList.get(inst)));
133             }
134         }
135         return processMap;
136     }

138     /**
139      * Class to encapsulate data from the registry performance counter block
140      */
141     @Immutable
142     public static class PerfCounterBlock {
143         private final String name;
144         private final int parentProcessID;
145         private final int priority;
146         private final long residentSetSize;
147         private final long startTime;
148         private final long upTime;
149         private final long bytesRead;
150         private final long bytesWritten;
151 
152         public PerfCounterBlock(String name, int parentProcessID, int priority, long residentSetSize, long startTime,
153                 long upTime, long bytesRead, long bytesWritten) {
154             this.name = name;
155             this.parentProcessID = parentProcessID;
156             this.priority = priority;
157             this.residentSetSize = residentSetSize;
158             this.startTime = startTime;
159             this.upTime = upTime;
160             this.bytesRead = bytesRead;
161             this.bytesWritten = bytesWritten;
162         }
163 
164         /**
165          * @return the name
166          */
167         public String getName() {
168             return name;
169         }
170 
171         /**
172          * @return the parentProcessID
173          */
174         public int getParentProcessID() {
175             return parentProcessID;
176         }
177 
178         /**
179          * @return the priority
180          */
181         public int getPriority() {
182             return priority;
183         }
184 
185         /**
186          * @return the residentSetSize
187          */
188         public long getResidentSetSize() {
189             return residentSetSize;
190         }
191 
192         /**
193          * @return the startTime
194          */
195         public long getStartTime() {
196             return startTime;
197         }
198 
199         /**
200          * @return the upTime
201          */
202         public long getUpTime() {
203             return upTime;
204         }
205 
206         /**
207          * @return the bytesRead
208          */
209         public long getBytesRead() {
210             return bytesRead;
211         }
212 
213         /**
214          * @return the bytesWritten
215          */
216         public long getBytesWritten() {
217             return bytesWritten;
218         }
219     }
