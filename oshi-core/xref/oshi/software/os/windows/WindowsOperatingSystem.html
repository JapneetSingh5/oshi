150     @Override
151     public FamilyVersionInfo queryFamilyVersionInfo() {
152         WmiResult<OSVersionProperty> versionInfo = Win32OperatingSystem.queryOsVersion();
153         if (versionInfo.getResultCount() < 1) {
154             return new FamilyVersionInfo("Windows", new OSVersionInfo(System.getProperty("os.version"), null, null));
155         }
156         // Guaranteed that versionInfo is not null and lists non-empty
157         // before calling the parse*() methods
158         int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
159         String buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0);
160         String version = parseVersion(versionInfo, suiteMask, buildNumber);
161         String codeName = parseCodeName(suiteMask);
162         return new FamilyVersionInfo("Windows", new OSVersionInfo(version, codeName, buildNumber));
163     }

165     private static String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask, String buildNumber) {
166         // Initialize a default, sane value
167         String version = System.getProperty("os.version");
168 
169         // Version is major.minor.build. Parse the version string for
170         // major/minor and get the build number separately
171         String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.VERSION, 0).split("\\D");
172         int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
173         int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
174 
175         // see
176         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
177         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.PRODUCTTYPE,
178                 0) == WinNT.VER_NT_WORKSTATION;
179         switch (major) {
180         case 10:
181             if (minor == 0) {
182                 if (ntWorkstation) {
183                     version = "10";
184                 } else {
185                     // Build numbers greater than 17762 is Server 2019 for OS
186                     // Version 10.0
187                     version = (ParseUtil.parseLongOrDefault(buildNumber, 0L) > 17762) ? "Server 2019" : "Server 2016";
188                 }
189             }
190             break;
191         case 6:
192             if (minor == 3) {
193                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
194             } else if (minor == 2) {
195                 version = ntWorkstation ? "8" : "Server 2012";
196             } else if (minor == 1) {
197                 version = ntWorkstation ? "7" : "Server 2008 R2";
198             } else if (minor == 0) {
199                 version = ntWorkstation ? "Vista" : "Server 2008";
200             }
201             break;
202         case 5:
203             if (minor == 2) {
204                 if ((suiteMask & 0x00008000) != 0) {// VER_SUITE_WH_SERVER
205                     version = "Home Server";
206                 } else if (ntWorkstation) {
207                     version = "XP"; // 64 bits
208                 } else {
209                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
210                             : "Server 2003 R2";
211                 }
212             } else if (minor == 1) {
213                 version = "XP"; // 32 bits
214             } else if (minor == 0) {
215                 version = "2000";
216             }
217             break;
218         default:
219             break;
220         }
221 
222         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
223         if (!sp.isEmpty() && !"unknown".equals(sp)) {
224             version = version + " " + sp.replace("Service Pack ", "SP");
225         }
226 
227         return version;
228     }

230     /**
231      * Gets suites available on the system and return as a codename
232      *
233      * @param suiteMask
234      *
235      * @return Suites
236      */
237     private static String parseCodeName(int suiteMask) {
238         List<String> suites = new ArrayList<>();
239         if ((suiteMask & 0x00000002) != 0) {
240             suites.add("Enterprise");
241         }
242         if ((suiteMask & 0x00000004) != 0) {
243             suites.add("BackOffice");
244         }
245         if ((suiteMask & 0x00000008) != 0) {
246             suites.add("Communication Server");
247         }
248         if ((suiteMask & 0x00000080) != 0) {
249             suites.add("Datacenter");
250         }
251         if ((suiteMask & 0x00000200) != 0) {
252             suites.add("Home");
253         }
254         if ((suiteMask & 0x00000400) != 0) {
255             suites.add("Web Server");
256         }
257         if ((suiteMask & 0x00002000) != 0) {
258             suites.add("Storage Server");
259         }
260         if ((suiteMask & 0x00004000) != 0) {
261             suites.add("Compute Cluster");
262         }
263         // 0x8000, Home Server, is included in main version name
264         return String.join(",", suites);
265     }

318     @Override
319     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
320         Set<Integer> childPids = new HashSet<>();
321         // Get processes from ToolHelp API for parent PID
322         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
323         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
324         try {
325             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
326                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
327                     childPids.add(processEntry.th32ProcessID.intValue());
328                 }
329             }
330         } finally {
331             Kernel32.INSTANCE.CloseHandle(snapshot);
332         }
333         List<OSProcess> procList = getProcesses(childPids);
334         List<OSProcess> sorted = processSort(procList, limit, sort);
335         return sorted.toArray(new OSProcess[0]);
336     }

344     /**
345      * Private method to do the heavy lifting for all the getProcess functions.
346      *
347      * @param pids
348      *            A collection of pids to query. If null, the entire process list
349      *            will be queried.
350      * @param slowFields
351      *            Whether to include fields that incur processor latency
352      * @return A corresponding list of processes
353      */
354     private List<OSProcess> processMapToList(Collection<Integer> pids, boolean slowFields) {
355         // Get data from the registry if possible
356         Map<Integer, OSProcess> processMap = processMapFromRegistry.get();
357         // otherwise performance counters with WMI backup
358         if (processMap == null) {
359             processMap = (pids == null) ? processMapFromPerfCounters.get() : buildProcessMapFromPerfCounters(pids);
360         }
361 
362         int myPid = getProcessId();
363 
364         // Structure we'll fill from native memory pointer for Vista+
365         Pointer pProcessInfo = null;
366         WTS_PROCESS_INFO_EX[] processInfo = null;
367         IntByReference pCount = new IntByReference(0);
368 
369         // WMI result we'll use for pre-Vista
370         WmiResult<ProcessXPProperty> processWmiResult = null;
371 
372         // Get processes from WTS (post-XP)
373         if (IS_WINDOWS7_OR_GREATER) {
374             final PointerByReference ppProcessInfo = new PointerByReference();
375             if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
376                     new IntByReference(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1), Wtsapi32.WTS_ANY_SESSION, ppProcessInfo,
377                     pCount)) {
378                 LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
379                 return new ArrayList<>(0);
380             }
381             // extract the pointed-to pointer and create array
382             pProcessInfo = ppProcessInfo.getValue();
383             final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
384             processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
385         } else {
386             // Pre-Vista we can't use WTSEnumerateProcessesEx so we'll grab the
387             // same info from WMI and fake the array
388             processWmiResult = Win32Process.queryProcesses(pids);
389         }
390 
391         // Store a subset of processes in a list to later return.
392         List<OSProcess> processList = new ArrayList<>();
393 
394         int procCount = IS_WINDOWS7_OR_GREATER ? processInfo.length : processWmiResult.getResultCount();
395         for (int i = 0; i < procCount; i++) {
396             int pid = IS_WINDOWS7_OR_GREATER ? processInfo[i].ProcessId
397                     : WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PROCESSID, i);
398             OSProcess proc;
399             // If the cache is empty, there was a problem with
400             // filling the cache using performance information.
401             if (processMap.isEmpty()) {
402                 if (pids != null && !pids.contains(pid)) {
403                     continue;
404                 }
405                 proc = new OSProcess(this);
406                 proc.setProcessID(pid);
407                 proc.setName(IS_WINDOWS7_OR_GREATER ? processInfo[i].pProcessName
408                         : WmiUtil.getString(processWmiResult, ProcessXPProperty.NAME, i));
409             } else {
410                 proc = processMap.get(pid);
411                 if (proc == null || pids != null && !pids.contains(pid)) {
412                     continue;
413                 }
414             }
415             // For my own process, set CWD
416             if (pid == myPid) {
417                 String cwd = new File(".").getAbsolutePath();
418                 // trim off trailing "."
419                 proc.setCurrentWorkingDirectory(cwd.isEmpty() ? "" : cwd.substring(0, cwd.length() - 1));
420             }
421 
422             if (IS_WINDOWS7_OR_GREATER) {
423                 WTS_PROCESS_INFO_EX procInfo = processInfo[i];
424                 proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
425                 proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
426                 proc.setThreadCount(procInfo.NumberOfThreads);
427                 proc.setVirtualSize(procInfo.PagefileUsage & 0xffff_ffffL);
428                 proc.setOpenFiles(procInfo.HandleCount);
429             } else {
430                 proc.setKernelTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.KERNELMODETIME, i) / 10000L);
431                 proc.setUserTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.USERMODETIME, i) / 10000L);
432                 proc.setThreadCount(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.THREADCOUNT, i));
433                 // WMI Pagefile usage is in KB
434                 proc.setVirtualSize(1024
435                         * (WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PAGEFILEUSAGE, i) & 0xffff_ffffL));
436                 proc.setOpenFiles(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.HANDLECOUNT, i));
437             }
438 
439             // Get a handle to the process for various extended info. Only gets
440             // current user unless running as administrator
441             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
442                     proc.getProcessID());
443             if (pHandle != null) {
444                 proc.setBitness(this.getBitness());
445                 // Only test for 32-bit process on 64-bit windows
446                 if (IS_VISTA_OR_GREATER && this.getBitness() == 64) {
447                     IntByReference wow64 = new IntByReference(0);
448                     if (Kernel32.INSTANCE.IsWow64Process(pHandle, wow64)) {
449                         proc.setBitness(wow64.getValue() > 0 ? 32 : 64);
450                     }
451                 }
452                 // Full path
453                 final HANDLEByReference phToken = new HANDLEByReference();
454                 try {// EXECUTABLEPATH
455                     proc.setPath(IS_WINDOWS7_OR_GREATER ? Kernel32Util.QueryFullProcessImageName(pHandle, 0)
456                             : WmiUtil.getString(processWmiResult, ProcessXPProperty.EXECUTABLEPATH, i));
457                     if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY,
458                             phToken)) {
459                         Account account = Advapi32Util.getTokenAccount(phToken.getValue());
460                         proc.setUser(account.name);
461                         proc.setUserID(account.sidString);
462                         // Fetching group information incurs ~10ms per process.
463                         if (slowFields) {
464                             Account a = oshi.jna.platform.windows.Advapi32Util.getTokenPrimaryGroup(phToken.getValue());
465                             proc.setGroup(a.name);
466                             proc.setGroupID(a.sidString);
467                         }
468                     } else {
469                         int error = Kernel32.INSTANCE.GetLastError();
470                         // Access denied errors are common. Fail silently.
471                         if (error != WinError.ERROR_ACCESS_DENIED) {
472                             LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
473                                     Kernel32.INSTANCE.GetLastError());
474                         }
475                     }
476                 } catch (Win32Exception e) {
477                     handleWin32ExceptionOnGetProcessInfo(proc, e);
478                 } finally {
479                     final HANDLE token = phToken.getValue();
480                     if (token != null) {
481                         Kernel32.INSTANCE.CloseHandle(token);
482                     }
483                 }
484                 Kernel32.INSTANCE.CloseHandle(pHandle);
485             }
486 
487             // There is no easy way to get ExecutuionState for a process.
488             // The WMI value is null. It's possible to get thread Execution
489             // State and possibly roll up.
490             proc.setState(OSProcess.State.RUNNING);
491 
492             // Initialize default
493             proc.setCommandLine("");
494 
495             processList.add(proc);
496         }
497         // Clean up memory allocated in C (only Vista+ but null pointer
498         // effectively tests)
499         if (pProcessInfo != null && !Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1, pProcessInfo,
500                 pCount.getValue())) {
501             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
502             return new ArrayList<>(0);
503         }
504 
505         // Command Line only accessible via WMI.
506         if (slowFields) {
507             Set<Integer> pidsToQuery = new HashSet<>();
508             if (pids != null) {
509                 for (OSProcess process : processList) {
510                     pidsToQuery.add(process.getProcessID());
511                 }
512             }
513             WmiResult<CommandLineProperty> commandLineProcs = Win32Process.queryCommandLines(pidsToQuery);
514             for (int p = 0; p < commandLineProcs.getResultCount(); p++) {
515                 int pid = WmiUtil.getUint32(commandLineProcs, CommandLineProperty.PROCESSID, p);
516                 if (processMap.containsKey(pid)) {
517                     OSProcess proc = processMap.get(pid);
518                     proc.setCommandLine(WmiUtil.getString(commandLineProcs, CommandLineProperty.COMMANDLINE, p));
519                 }
520             }
521         }
522         return processList;
523     }

528     }

538     private Map<Integer, OSProcess> buildProcessMapFromPerfCounters(Collection<Integer> pids) {
539         Map<Integer, OSProcess> processMap = new HashMap<>();
540         Pair<List<String>, Map<ProcessPerformanceProperty, List<Long>>> instanceValues = ProcessInformation
541                 .queryProcessCounters();
542         long now = System.currentTimeMillis(); // 1970 epoch
543         List<String> instances = instanceValues.getA();
544         Map<ProcessPerformanceProperty, List<Long>> valueMap = instanceValues.getB();
545         List<Long> pidList = valueMap.get(ProcessPerformanceProperty.PROCESSID);
546         List<Long> ppidList = valueMap.get(ProcessPerformanceProperty.PARENTPROCESSID);
547         List<Long> priorityList = valueMap.get(ProcessPerformanceProperty.PRIORITY);
548         List<Long> ioReadList = valueMap.get(ProcessPerformanceProperty.READTRANSFERCOUNT);
549         List<Long> ioWriteList = valueMap.get(ProcessPerformanceProperty.WRITETRANSFERCOUNT);
550         List<Long> workingSetSizeList = valueMap.get(ProcessPerformanceProperty.PRIVATEPAGECOUNT);
551         List<Long> creationTimeList = valueMap.get(ProcessPerformanceProperty.CREATIONDATE);
552 
553         for (int inst = 0; inst < instances.size(); inst++) {
554             int pid = pidList.get(inst).intValue();
555             if (pids == null || pids.contains(pid)) {
556                 OSProcessml#OSProcess">OSProcess proc = new OSProcess(this);
557                 processMap.put(pid, proc);
558 
559                 proc.setProcessID(pid);
560                 proc.setName(instances.get(inst));
561                 proc.setParentProcessID(ppidList.get(inst).intValue());
562                 proc.setPriority(priorityList.get(inst).intValue());
563                 // if creation time value is less than current millis, it's in 1970 epoch,
564                 // otherwise it's 1601 epoch and we must convert
565                 long ctime = creationTimeList.get(inst);
566                 if (ctime > now) {
567                     ctime = WinBase.FILETIME.filetimeToDate((int) (ctime >> 32), (int) (ctime & 0xffffffffL)).getTime();
568                 }
569                 proc.setUpTime(now - ctime);
570                 proc.setStartTime(ctime);
571                 proc.setBytesRead(ioReadList.get(inst));
572                 proc.setBytesWritten(ioWriteList.get(inst));
573                 proc.setResidentSetSize(workingSetSizeList.get(inst));
574             }
575         }
576         return processMap;
577     }

622     private static long querySystemUptime() {
623         // Uptime is in seconds so divide milliseconds
624         // GetTickCount64 requires Vista (6.0) or later
625         if (IS_VISTA_OR_GREATER) {
626             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
627         } else {
628             // 32 bit rolls over at ~ 49 days
629             return Kernel32.INSTANCE.GetTickCount() / 1000L;
630         }

638     private static long querySystemBootTime() {
639         String eventLog = systemLog.get();
640         if (eventLog != null) {
641             try {
642                 EventLogIterator iter = new EventLogIterator(null, eventLog, WinNT.EVENTLOG_BACKWARDS_READ);
643                 // Get the most recent boot event (ID 12) from the Event log. If Windows "Fast
644                 // Startup" is enabled we may not see event 12, so also check for most recent ID
645                 // 6005 (Event log startup) as a reasonably close backup.
646                 long event6005Time = 0L;
647                 while (iter.hasNext()) {
648                     EventLogRecord record = iter.next();
649                     if (record.getStatusCode() == 12) {
650                         // Event 12 is system boot. We want this value unless we find two 6005 events
651                         // first (may occur with Fast Boot)
652                         return record.getRecord().TimeGenerated.longValue();
653                     } else if (record.getStatusCode() == 6005) {
654                         // If we already found one, this means we've found a second one without finding
655                         // an event 12. Return the latest one.
656                         if (event6005Time > 0) {
657                             return event6005Time;
658                         }
659                         // First 6005; tentatively assign
660                         event6005Time = record.getRecord().TimeGenerated.longValue();
661                     }
662                 }
663                 // Only one 6005 found, return
664                 if (event6005Time > 0) {
665                     return event6005Time;
666                 }
667             } catch (Win32Exception e) {
668                 LOG.warn("Can't open event log \"{}\".", eventLog);
669             }
670         }
671         // If we get this far, event log reading has failed, either from no log or no
672         // startup times. Subtract up time from current time as a reasonable proxy.
673         return System.currentTimeMillis() / 1000L - querySystemUptime();
674     }

681     /**
682      * Enables debug privileges for this process, required for OpenProcess() to get
683      * processes other than the current user
684      */
685     private static void enableDebugPrivilege() {
686         HANDLEByReference hToken = new HANDLEByReference();
687         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
688                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
689         if (!success) {
690             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
691             return;
692         }
693         WinNT.LUID luid = new WinNT.LUID();
694         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
695         if (!success) {
696             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
697             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
698             return;
699         }
700         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
701         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
702         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
703         if (!success) {
704             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
705         }
706         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
707     }

709     @Override
710     public OSService[] getServices() {
711         try (W32ServiceManager sm = new W32ServiceManager()) {
712             sm.open(Winsvc.SC_MANAGER_ENUMERATE_SERVICE);
713             Winsvc.ENUM_SERVICE_STATUS_PROCESS[] services = sm.enumServicesStatusExProcess(WinNT.SERVICE_WIN32,
714                     Winsvc.SERVICE_STATE_ALL, null);
715             OSServiceervice">OSService[] svcArray = new OSService[services.length];
716             for (int i = 0; i < services.length; i++) {
717                 State state;
718                 switch (services[i].ServiceStatusProcess.dwCurrentState) {
719                 case 1:
720                     state = STOPPED;
721                     break;
722                 case 4:
723                     state = RUNNING;
724                     break;
725                 default:
726                     state = OTHER;
727                     break;
728                 }
729                 svcArray[i] = new OSService(services[i].lpDisplayName, services[i].ServiceStatusProcess.dwProcessId,
730                         state);
731             }
732             return svcArray;
733         } catch (com.sun.jna.platform.win32.Win32Exception ex) {
734             LOG.error("Win32Exception: {}", ex.getMessage());
735             return new OSService[0];
736         }
737     }

739     private static String querySystemLog() {
740         String systemLog = GlobalConfig.get("oshi.os.windows.eventlog", "System");
741         if (systemLog.isEmpty()) {
742             // Use faster boot time approximation
743             return null;
744         }
745         // Check whether it works
746         HANDLE h = Advapi32.INSTANCE.OpenEventLog(null, systemLog);
747         if (h == null) {
748             LOG.warn("Unable to open configured system Event log \"{}\". Calculating boot time from uptime.",
749                     systemLog);
750             return null;
751         }
752         return systemLog;
753     }
