122     // Get a list of orders to pass to ParseUtil
123     private static final int[] PROC_PID_STAT_ORDERS = new int[ProcPidStat.values().length];
124     static {
125         for (ProcPidStat stat : ProcPidStat.values()) {
126             // The PROC_PID_STAT enum indices are 1-indexed.
127             // Subtract one to get a zero-based index
128             PROC_PID_STAT_ORDERS[stat.ordinal()] = stat.getOrder() - 1;
129         }
130     }

132     // 2.6 Kernel has 44 elements, 3.3 has 47, and 3.5 has 52.
133     // Check /proc/self/stat to find its length
134     private static final int PROC_PID_STAT_LENGTH;
135     static {
136         String stat = FileUtil.getStringFromFile(ProcPath.SELF_STAT);
137         if (!stat.isEmpty() && stat.contains(")")) {
138             // add 3 to account for pid, process name in prarenthesis, and state
139             PROC_PID_STAT_LENGTH = ParseUtil.countStringToLongArray(stat, ' ') + 3;
140         } else {
141             // Default assuming recent kernel
142             PROC_PID_STAT_LENGTH = 52;
143         }
144     }

146     // Jiffies per second, used for process time counters.
147     private static final long USER_HZ = ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf CLK_TCK"),
148             100L);
149     // Boot time in MS.
150     private static final long BOOT_TIME;
151     static {
152         // Uptime is only in hundredths of seconds but we need thousandths.
153         // We can grab uptime twice and take average to reduce error, getting
154         // current time in between
155         double uptime = UpTime.getSystemUptimeSeconds();
156         long now = System.currentTimeMillis();
157         uptime += UpTime.getSystemUptimeSeconds();
158         // Uptime is now 2x seconds, so divide by 2, but
159         // we want milliseconds so multiply by 1000
160         // Ultimately multiply by 1000/2 = 500
161         BOOT_TIME = now - (long) (500d * uptime + 0.5);
162         // Cast/truncation is effectively rounding. Boot time could
163         // be +/- 5 ms due to System Uptime rounding to nearest 10ms.
164     }

256     private OSProcess getProcess(int pid, LinuxUserGroupInfo userGroupInfo, boolean slowFields) {
257         String path = "";
258         String procPidExe = String.format(ProcPath.PID_EXE, pid);
259         try {
260             Path link = Paths.get(procPidExe);
261             path = Files.readSymbolicLink(link).toString();
262             int index = path.indexOf(" (deleted)");
263             if (index != -1) {
264                 path = path.substring(0, index);
265             }
266         } catch (InvalidPathException | IOException | UnsupportedOperationException | SecurityException e) {
267             LOG.debug("Unable to open symbolic link {}", procPidExe);
268         }
269         Map<String, String> io = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_IO, pid), ":");
270         // See man proc for how to parse /proc/[pid]/stat
271         long now = System.currentTimeMillis();
272         String stat = FileUtil.getStringFromFile(String.format(ProcPath.PID_STAT, pid));
273         // A race condition may leave us with an empty string
274         if (stat.isEmpty()) {
275             return null;
276         }
277         // We can get name and status more easily from /proc/pid/status which we
278         // call later, so just get the numeric bits here
279         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
280         // Fetch cached process if it exists
281         OSProcessml#OSProcess">OSProcess proc = new OSProcess(this);
282         proc.setProcessID(pid);
283         // The /proc/pid/cmdline value is null-delimited
284         proc.setCommandLine(FileUtil.getStringFromFile(String.format(ProcPath.PID_CMDLINE, pid)));
285         // BOOTTIME is in seconds and start time from proc/pid/stat is in jiffies.
286         // Combine units to jiffies and convert to millijiffies before hz division to
287         // avoid precision loss without having to cast
288         long startTime = (BOOT_TIME * USER_HZ + statArray[ProcPidStat.START_TIME.ordinal()]) * 1000L / USER_HZ;
289         // BOOT_TIME could be up to 500ms off and start time up to 5ms off. A process
290         // that has started within last 505ms could produce a future start time/negative
291         // up time, so insert a sanity check.
292         if (startTime >= now) {
293             startTime = now - 1;
294         }
295         proc.setStartTime(startTime);
296         proc.setParentProcessID((int) statArray[ProcPidStat.PPID.ordinal()]);
297         proc.setThreadCount((int) statArray[ProcPidStat.THREAD_COUNT.ordinal()]);
298         proc.setPriority((int) statArray[ProcPidStat.PRIORITY.ordinal()]);
299         proc.setVirtualSize(statArray[ProcPidStat.VSZ.ordinal()]);
300         proc.setResidentSetSize(statArray[ProcPidStat.RSS.ordinal()] * this.memoryPageSize);
301         proc.setKernelTime(statArray[ProcPidStat.KERNEL_TIME.ordinal()] * 1000L / USER_HZ);
302         proc.setUserTime(statArray[ProcPidStat.USER_TIME.ordinal()] * 1000L / USER_HZ);
303         proc.setUpTime(now - proc.getStartTime());
304         // See man proc for how to parse /proc/[pid]/io
305         proc.setBytesRead(ParseUtil.parseLongOrDefault(io.getOrDefault("read_bytes", ""), 0L));
306         proc.setBytesWritten(ParseUtil.parseLongOrDefault(io.getOrDefault("write_bytes", ""), 0L));
307 
308         // gets the open files count
309         if (slowFields) {
310             List<String> openFilesList = ExecutingCommand.runNative(String.format(LS_F_PROC_PID_FD, pid));
311             proc.setOpenFiles(openFilesList.size() - 1L);
312 
313             // get 5th byte of file for 64-bit check
314             // https://en.wikipedia.org/wiki/Executable_and_Linkable_Format#File_header
315             byte[] buffer = new byte[5];
316             if (!path.isEmpty()) {
317                 try (InputStream is = new FileInputStream(path)) {
318                     if (is.read(buffer) == buffer.length) {
319                         proc.setBitness(buffer[4] == 1 ? 32 : 64);
320                     }
321                 } catch (IOException e) {
322                     LOG.warn("Failed to read process file: {}", path);
323                 }
324             }
325         }
326 
327         Map<String, String> status = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_STATUS, pid), ":");
328         proc.setName(status.getOrDefault("Name", ""));
329         proc.setPath(path);
330         switch (status.getOrDefault("State", "U").charAt(0)) {
331         case 'R':
332             proc.setState(OSProcess.State.RUNNING);
333             break;
334         case 'S':
335             proc.setState(OSProcess.State.SLEEPING);
336             break;
337         case 'D':
338             proc.setState(OSProcess.State.WAITING);
339             break;
340         case 'Z':
341             proc.setState(OSProcess.State.ZOMBIE);
342             break;
343         case 'T':
344             proc.setState(OSProcess.State.STOPPED);
345             break;
346         default:
347             proc.setState(OSProcess.State.OTHER);
348             break;
349         }
350         proc.setUserID(ParseUtil.whitespaces.split(status.getOrDefault("Uid", ""))[0]);
351         proc.setGroupID(ParseUtil.whitespaces.split(status.getOrDefault("Gid", ""))[0]);
352         Pair<String, String> user = userGroupInfo.getUser(proc.getUserID());
353         proc.setUser(user.getB());
354         proc.setGroup(userGroupInfo.getGroupName(proc.getGroupID()));
355 
356         try {
357             String cwdLink = String.format(ProcPath.PID_CWD, pid);
358             String cwd = new File(cwdLink).getCanonicalPath();
359             if (!cwd.equals(cwdLink)) {
360                 proc.setCurrentWorkingDirectory(cwd);
361             }
362         } catch (IOException e) {
363             LOG.trace("Couldn't find cwd for pid {}: {}", pid, e.getMessage());
364         }
365         return proc;
366     }

370         List<OSProcess> procs = new ArrayList<>();
371         File[] procFiles = ProcessStat.getPidFiles();
372         LinuxUserGroupInfoLinuxUserGroupInfo">LinuxUserGroupInfo userGroupInfo = new LinuxUserGroupInfo();

374         // now for each file (with digit name) get process info
375         for (File procFile : procFiles) {

378                 OSProcess proc = getProcess(pid, userGroupInfo, true);
379                 if (proc != null) {

381                 }
382             }
383         }
384         List<OSProcess> sorted = processSort(procs, limit, sort);
385         return sorted.toArray(new OSProcess[0]);
386     }

388     private static int getParentPidFromProcFile(int pid) {
389         String stat = FileUtil.getStringFromFile(String.format("/proc/%d/stat", pid));
390         // A race condition may leave us with an empty string
391         if (stat.isEmpty()) {
392             return 0;
393         }
394         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
395         return (int) statArray[ProcPidStat.PPID.ordinal()];
396     }

398     @Override
399     public long getProcessAffinityMask(int processId) {
400         // Would prefer to use native sched_getaffinity call but variable sizing is
401         // kernel-dependent and requires C macros, so we use command line instead.
402         String mask = ExecutingCommand.getFirstAnswer("taskset -p " + processId);
403         // Output:
404         // pid 3283's current affinity mask: 3
405         // pid 9726's current affinity mask: f
406         String[] split = ParseUtil.whitespaces.split(mask);
407         try {
408             return new BigInteger(split[split.length - 1], 16).longValue();
409         } catch (NumberFormatException e) {
410             return 0;
411         }
412     }

454     private static Triplet<String, String, String> queryFamilyVersionCodenameFromReleaseFiles() {
455         Triplet<String, String, String> familyVersionCodename;
456         // There are two competing options for family/version information.
457         // Newer systems are adopting a standard /etc/os-release file:
458         // https://www.freedesktop.org/software/systemd/man/os-release.html
459         //
460         // Some systems are still using the lsb standard which parses a
461         // variety of /etc/*-release files and is most easily accessed via
462         // the commandline lsb_release -a, see here:
463         // http://linux.die.net/man/1/lsb_release
464         // In this case, the /etc/lsb-release file (if it exists) has
465         // optional overrides to the information in the /etc/distrib-release
466         // files, which show: "Distributor release x.x (Codename)"
467 
468         // Attempt to read /etc/system-release which has more details than
469         // os-release on (CentOS and Fedora)
470         if ((familyVersionCodename = readDistribRelease("/etc/system-release")) != null) {
471             // If successful, we're done. this.family has been set and
472             // possibly the versionID and codeName
473             return familyVersionCodename;
474         }
475 
476         // Attempt to read /etc/os-release file.
477         if ((familyVersionCodename = readOsRelease()) != null) {
478             // If successful, we're done. this.family has been set and
479             // possibly the versionID and codeName
480             return familyVersionCodename;
481         }
482 
483         // Attempt to execute the `lsb_release` command
484         if ((familyVersionCodename = execLsbRelease()) != null) {
485             // If successful, we're done. this.family has been set and
486             // possibly the versionID and codeName
487             return familyVersionCodename;
488         }
489 
490         // The above two options should hopefully work on most
491         // distributions. If not, we keep having fun.
492         // Attempt to read /etc/lsb-release file
493         if ((familyVersionCodename = readLsbRelease()) != null) {
494             // If successful, we're done. this.family has been set and
495             // possibly the versionID and codeName
496             return familyVersionCodename;
497         }
498 
499         // If we're still looking, we search for any /etc/*-release (or
500         // similar) filename, for which the first line should be of the
501         // "Distributor release x.x (Codename)" format or possibly a
502         // "Distributor VERSION x.x (Codename)" format
503         String etcDistribRelease = getReleaseFilename();
504         if ((familyVersionCodename = readDistribRelease(etcDistribRelease)) != null) {
505             // If successful, we're done. this.family has been set and
506             // possibly the versionID and codeName
507             return familyVersionCodename;
508         }
509         // If we've gotten this far with no match, use the distrib-release
510         // filename (defaults will eventually give "Unknown")
511         String family = filenameToFamily(etcDistribRelease.replace("/etc/", "").replace("release", "")
512                 .replace("version", "").replace("-", "").replace("_", ""));
513         return new Triplet<>(family, Constants.UNKNOWN, Constants.UNKNOWN);
514     }

516     /**
517      * Attempts to read /etc/os-release
518      *
519      * @return a triplet with the parsed family, versionID and codeName if file
520      *         successfully read and NAME= found, null otherwise
521      */
522     private static Triplet<String, String, String> readOsRelease() {
523         String family = null;
524         String versionId = Constants.UNKNOWN;
525         String codeName = Constants.UNKNOWN;
526         List<String> osRelease = FileUtil.readFile("/etc/os-release");
527         // Search for NAME=
528         for (String line : osRelease) {
529             if (line.startsWith("VERSION=")) {
530                 LOG.debug(OS_RELEASE_LOG, line);
531                 // remove beginning and ending '"' characters, etc from
532                 // VERSION="14.04.4 LTS, Trusty Tahr" (Ubuntu style)
533                 // or VERSION="17 (Beefy Miracle)" (os-release doc style)
534                 line = line.replace("VERSION=", "").replaceAll(DOUBLE_QUOTES, "").trim();
535                 String[] split = line.split("[()]");
536                 if (split.length <= 1) {
537                     // If no parentheses, check for Ubuntu's comma format
538                     split = line.split(", ");
539                 }
540                 if (split.length > 0) {
541                     versionId = split[0].trim();
542                 }
543                 if (split.length > 1) {
544                     codeName = split[1].trim();
545                 }
546             } else if (line.startsWith("NAME=") && family == null) {
547                 LOG.debug(OS_RELEASE_LOG, line);
548                 // remove beginning and ending '"' characters, etc from
549                 // NAME="Ubuntu"
550                 family = line.replace("NAME=", "").replaceAll(DOUBLE_QUOTES, "").trim();
551             } else if (line.startsWith("VERSION_ID=") && versionId.equals(Constants.UNKNOWN)) {
552                 LOG.debug(OS_RELEASE_LOG, line);
553                 // remove beginning and ending '"' characters, etc from
554                 // VERSION_ID="14.04"
555                 versionId = line.replace("VERSION_ID=", "").replaceAll(DOUBLE_QUOTES, "").trim();
556             }
557         }
558         return family == null ? null : new Triplet<>(family, versionId, codeName);
559     }

561     /**
562      * Attempts to execute `lsb_release -a`
563      *
564      * @return a triplet with the parsed family, versionID and codeName if the
565      *         command successfully executed and Distributor ID: or Description:
566      *         found, null otherwise
567      */
568     private static Triplet<String, String, String> execLsbRelease() {
569         String family = null;
570         String versionId = Constants.UNKNOWN;
571         String codeName = Constants.UNKNOWN;
572         // If description is of the format Distrib release x.x (Codename)
573         // that is primary, otherwise use Distributor ID: which returns the
574         // distribution concatenated, e.g., RedHat instead of Red Hat
575         for (String line : ExecutingCommand.runNative("lsb_release -a")) {
576             if (line.startsWith("Description:")) {
577                 LOG.debug(LSB_RELEASE_A_LOG, line);
578                 line = line.replace("Description:", "").trim();
579                 if (line.contains(RELEASE_DELIM)) {
580                     Triplet<String, String, String> triplet = parseRelease(line, RELEASE_DELIM);
581                     family = triplet.getA();
582                     if (versionId.equals(Constants.UNKNOWN)) {
583                         versionId = triplet.getB();
584                     }
585                     if (codeName.equals(Constants.UNKNOWN)) {
586                         codeName = triplet.getC();
587                     }
588                 }
589             } else if (line.startsWith("Distributor ID:") && family == null) {

647      * @return a triplet with the parsed family, versionID and codeName if file
648      *         successfully read and " release " or " VERSION " found, null
649      *         otherwise
650      */
651     private static Triplet<String, String, String> readDistribRelease(String filename) {
652         if (new File(filename).exists()) {
653             List<String> osRelease = FileUtil.readFile(filename);
654             // Search for Distrib release x.x (Codename)
655             for (String line : osRelease) {
656                 LOG.debug("{}: {}", filename, line);
657                 if (line.contains(RELEASE_DELIM)) {
658                     // If this parses properly we're done
659                     return parseRelease(line, RELEASE_DELIM);
660                 } else if (line.contains(" VERSION ")) {
661                     // If this parses properly we're done
662                     return parseRelease(line, " VERSION ");
663                 }
664             }
665         }
666         return null;

669     /**
670      * Helper method to parse version description line style
671      *
672      * @param line
673      *            a String of the form "Distributor release x.x (Codename)"
674      * @param splitLine
675      *            A regex to split on, e.g. " release "
676      * @return a triplet with the parsed family, versionID and codeName
677      */
678     private static Triplet<String, String, String> parseRelease(String line, String splitLine) {
679         String[] split = line.split(splitLine);
680         String family = split[0].trim();
681         String versionId = Constants.UNKNOWN;
682         String codeName = Constants.UNKNOWN;
683         if (split.length > 1) {
684             split = split[1].split("[()]");
685             if (split.length > 0) {
686                 versionId = split[0].trim();
687             }
688             if (split.length > 1) {
689                 codeName = split[1].trim();
690             }
691         }
692         return new Triplet<>(family, versionId, codeName);
693     }

714         }
715         if (new File("/etc/release").exists()) {
716             return "/etc/release";
717         }
718         // If all else fails, try this
719         return "/etc/issue";
