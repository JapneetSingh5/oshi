62      // Get a list of orders to pass to ParseUtil
63      private static final int[] PROC_PID_STAT_ORDERS = new int[ProcPidStat.values().length];
64      static {
65          for (ProcPidStat stat : ProcPidStat.values()) {
66              // The PROC_PID_STAT enum indices are 1-indexed.
67              // Subtract one to get a zero-based index
68              PROC_PID_STAT_ORDERS[stat.ordinal()] = stat.getOrder() - 1;
69          }
70      }

115     @Override
116     public String getCurrentWorkingDirectory() {
117         try {
118             String cwdLink = String.format(ProcPath.PID_CWD, getProcessID());
119             String cwd = new File(cwdLink).getCanonicalPath();
120             if (!cwd.equals(cwdLink)) {
121                 return cwd;
122             }
123         } catch (IOException e) {
124             LOG.trace("Couldn't find cwd for pid {}: {}", getProcessID(), e.getMessage());
125         }
126         return "";
127     }

220     private int queryBitness() {
221         // get 5th byte of file for 64-bit check
222         // https://en.wikipedia.org/wiki/Executable_and_Linkable_Format#File_header
223         byte[] buffer = new byte[5];
224         if (!path.isEmpty()) {
225             try (InputStream is = new FileInputStream(path)) {
226                 if (is.read(buffer) == buffer.length) {
227                     return buffer[4] == 1 ? 32 : 64;
228                 }
229             } catch (IOException e) {
230                 LOG.warn("Failed to read process file: {}", path);
231             }
232         }
233         return 0;
234     }

236     @Override
237     public long getAffinityMask() {
238         // Would prefer to use native sched_getaffinity call but variable sizing is
239         // kernel-dependent and requires C macros, so we use command line instead.
240         String mask = ExecutingCommand.getFirstAnswer("taskset -p " + getProcessID());
241         // Output:
242         // pid 3283's current affinity mask: 3
243         // pid 9726's current affinity mask: f
244         String[] split = ParseUtil.whitespaces.split(mask);
245         try {
246             return new BigInteger(split[split.length - 1], 16).longValue();
247         } catch (NumberFormatException e) {
248             return 0;
249         }
250     }

252     @Override
253     public boolean updateAttributes() {
254         String procPidExe = String.format(ProcPath.PID_EXE, getProcessID());
255         try {
256             Path link = Paths.get(procPidExe);
257             this.path = Files.readSymbolicLink(link).toString();
258             // For some services the symbolic link process has terminated
259             int index = path.indexOf(" (deleted)");
260             if (index != -1) {
261                 path = path.substring(0, index);
262             }
263         } catch (InvalidPathException | IOException | UnsupportedOperationException | SecurityException e) {
264             LOG.debug("Unable to open symbolic link {}", procPidExe);
265         }
266         // Fetch all the values here
267         // check for terminated process race condition after last one.
268         Map<String, String> io = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_IO, getProcessID()), ":");
269         Map<String, String> status = FileUtil.getKeyValueMapFromFile(String.format(ProcPath.PID_STATUS, getProcessID()),
270                 ":");
271         String stat = FileUtil.getStringFromFile(String.format(ProcPath.PID_STAT, getProcessID()));
272         if (stat.isEmpty()) {
273             this.state = State.INVALID;
274             return false;
275         }
276         long now = System.currentTimeMillis();
277 
278         // We can get name and status more easily from /proc/pid/status which we
279         // call later, so just get the numeric bits here
280         // See man proc for how to parse /proc/[pid]/stat
281         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS,
282                 ProcessStat.PROC_PID_STAT_LENGTH, ' ');
283 
284         // BOOTTIME is in seconds and start time from proc/pid/stat is in jiffies.
285         // Combine units to jiffies and convert to millijiffies before hz division to
286         // avoid precision loss without having to cast
287         this.startTime = (LinuxOperatingSystem.BOOTTIME * LinuxOperatingSystem.getHz()
288                 + statArray[ProcPidStat.START_TIME.ordinal()]) * 1000L / LinuxOperatingSystem.getHz();
289         // BOOT_TIME could be up to 500ms off and start time up to 5ms off. A process
290         // that has started within last 505ms could produce a future start time/negative
291         // up time, so insert a sanity check.
292         if (startTime >= now) {
293             startTime = now - 1;
294         }
295         this.parentProcessID = (int) statArray[ProcPidStat.PPID.ordinal()];
296         this.threadCount = (int) statArray[ProcPidStat.THREAD_COUNT.ordinal()];
297         this.priority = (int) statArray[ProcPidStat.PRIORITY.ordinal()];
298         this.virtualSize = statArray[ProcPidStat.VSZ.ordinal()];
299         this.residentSetSize = statArray[ProcPidStat.RSS.ordinal()] * PAGE_SIZE;
300         this.kernelTime = statArray[ProcPidStat.KERNEL_TIME.ordinal()] * 1000L / LinuxOperatingSystem.getHz();
301         this.userTime = statArray[ProcPidStat.USER_TIME.ordinal()] * 1000L / LinuxOperatingSystem.getHz();
302         this.upTime = now - startTime;
303 
304         // See man proc for how to parse /proc/[pid]/io
305         this.bytesRead = ParseUtil.parseLongOrDefault(io.getOrDefault("read_bytes", ""), 0L);
306         this.bytesWritten = ParseUtil.parseLongOrDefault(io.getOrDefault("write_bytes", ""), 0L);
307 
308         // Don't set open files or bitness or currentWorkingDirectory; fetch on demand.
309 
310         this.userID = ParseUtil.whitespaces.split(status.getOrDefault("Uid", ""))[0];
311         this.user = UserGroupInfo.getUser(userID);
312         this.groupID = ParseUtil.whitespaces.split(status.getOrDefault("Gid", ""))[0];
313         this.group = UserGroupInfo.getGroupName(groupID);
314         this.name = status.getOrDefault("Name", "");
315         switch (status.getOrDefault("State", "U").charAt(0)) {
316         case 'R':
317             this.state = State.RUNNING;
318             break;
319         case 'S':
320             this.state = State.SLEEPING;
321             break;
322         case 'D':
323             this.state = State.WAITING;
324             break;
325         case 'Z':
326             this.state = State.ZOMBIE;
327             break;
328         case 'T':
329             this.state = State.STOPPED;
330             break;
331         default:
332             this.state = State.OTHER;
333             break;
334         }
335         return true;
336     }

338     /**
339      * Enum used to update attributes. The order field represents the 1-indexed
340      * numeric order of the stat in /proc/pid/stat per the man file.
341      */
342     private enum ProcPidStat {
343         // The parsing implementation in ParseUtil requires these to be declared
344         // in increasing order
345         PPID(4), USER_TIME(14), KERNEL_TIME(15), PRIORITY(18), THREAD_COUNT(20), START_TIME(22), VSZ(23), RSS(24);
346 
347         private int order;
348 
349         public int getOrder() {
350             return this.order;
351         }
352 
353         ProcPidStat(int order) {
354             this.order = order;
355         }
356     }
