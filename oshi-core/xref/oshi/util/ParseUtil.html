153     /**
154      * Parse hertz from a string, eg. "2.00MHz" is 2000000L.
155      *
156      * @param hertz
157      *            Hertz size.
158      * @return {@link java.lang.Long} Hertz value or -1 if not parseable.
159      */
160     public static long parseHertz(String hertz) {
161         Matcher matcher = HERTZ_PATTERN.matcher(hertz.trim());
162         if (matcher.find() && matcher.groupCount() == 3) {
163             // Regexp enforces #(.#) format so no test for NFE required
164             double value = Double.valueOf(matcher.group(1)) * multipliers.getOrDefault(matcher.group(3), -1L);
165             if (value >= 0d) {
166                 return (long) value;
167             }
168         }
169         return -1L;
170     }

252     /**
253      * Parse a byte aray into a string of hexadecimal digits including leading zeros
254      *
255      * @param bytes
256      *            The byte array to represent
257      * @return A string of hex characters corresponding to the bytes. The string is
258      *         upper case.
259      */
260     public static String byteArrayToHexString(byte[] bytes) {
261         // Solution copied from https://stackoverflow.com/questions/9655181
262         char[] hexChars = new char[bytes.length * 2];
263         for (int j = 0; j < bytes.length; j++) {
264             int v = bytes[j] & 0xFF;
265             hexChars[j * 2] = HEX_ARRAY[v >>> 4];
266             hexChars[j * 2 + 1] = HEX_ARRAY[v & 0x0F];
267         }
268         return new String(hexChars);
269     }

271     /**
272      * Parse a string of hexadecimal digits into a byte array
273      *
274      * @param digits
275      *            The string to be parsed
276      * @return a byte array with each pair of characters converted to a byte, or
277      *         empty array if the string is not valid hex
278      */
279     public static byte[] hexStringToByteArray(String digits) {
280         int len = digits.length();
281         // Check if string is valid hex
282         if (!VALID_HEX.matcher(digits).matches() || (len & 0x1) != 0) {
283             LOG.warn("Invalid hexadecimal string: {}", digits);
284             return new byte[0];
285         }
286         byte[] data = new byte[len / 2];
287         for (int i = 0; i < len; i += 2) {
288             data[i / 2] = (byte) (Character.digit(digits.charAt(i), 16) << 4
289                     | Character.digit(digits.charAt(i + 1), 16));
290         }
291         return data;
292     }

310     /**
311      * Convert a long value to a byte array using Big Endian, truncating or padding
312      * with zeros (if necessary) so the array has the specified length.
313      *
314      * @param value
315      *            The value to be converted
316      * @param valueSize
317      *            Number of bytes representing the value
318      * @param length
319      *            Number of bytes to return
320      * @return A byte array of specified length representing the long in the first
321      *         valueSize bytes
322      */
323     public static byte[] longToByteArray(long value, int valueSize, int length) {
324         long val = value;
325         // Convert the long to 8-byte BE representation
326         byte[] b = new byte[8];
327         for (int i = 7; i >= 0 && val != 0L; i--) {
328             b[i] = (byte) val;
329             val >>>= 8;
330         }
331         // Then copy the rightmost valueSize bytes
332         // e.g., for an integer we want rightmost 4 bytes
333         return Arrays.copyOfRange(b, 8 - valueSize, 8 + length - valueSize);
334     }

350     /**
351      * Convert a byte array to its integer representation.
352      *
353      * @param bytes
354      *            An array of bytes no smaller than the size to be converted
355      * @param size
356      *            Number of bytes to convert to the long. May not exceed 8.
357      * @return An integer representing the byte array as a 64-bit number
358      */
359     public static long byteArrayToLong(byte[] bytes, int size) {
360         if (size > 8) {
361             throw new IllegalArgumentException("Can't convert more than 8 bytes.");
362         }
363         if (size > bytes.length) {
364             throw new IllegalArgumentException("Size can't be larger than array length.");
365         }
366         long total = 0L;
367         for (int i = 0; i < size; i++) {
368             total = total << 8 | bytes[i] & 0xff;
369         }
370         return total;
371     }

390     /**
391      * Convert an unsigned integer to a long value. The method assumes that all bits
392      * in the specified integer value are 'data' bits, including the
393      * most-significant bit which Java normally considers a sign bit. The method
394      * must be used only when it is certain that the integer value represents an
395      * unsigned integer, for example when the integer is returned by JNA library in
396      * a structure which holds unsigned integers.
397      *
398      * @param unsignedValue
399      *            The unsigned integer value to convert.
400      * @return The unsigned integer value widened to a long.
401      */
402     public static long unsignedIntToLong(int unsignedValue) {
403         // use standard Java widening conversion to long which does
404         // sign-extension,
405         // then drop any copies of the sign bit, to prevent the value being
406         // considered a negative one by Java if it is set
407         long longValue = unsignedValue;
408         return longValue & 0xffffffffL;
409     }

424     /**
425      * Parses a string of hex digits to a string where each pair of hex digits
426      * represents an ASCII character
427      *
428      * @param hexString
429      *            A sequence of hex digits
430      * @return The corresponding string if valid hex; otherwise the original
431      *         hexString
432      */
433     public static String hexStringToString(String hexString) {
434         // Odd length strings won't parse, return
435         if (hexString.length() % 2 > 0) {
436             return hexString;
437         }
438         int charAsInt;
439         StringBuilder sb = new StringBuilder();
440         try {
441             for (int pos = 0; pos < hexString.length(); pos += 2) {
442                 charAsInt = Integer.parseInt(hexString.substring(pos, pos + 2), 16);
443                 if (charAsInt < 32 || charAsInt > 127) {
444                     return hexString;
445                 }
446                 sb.append((char) charAsInt);
447             }
448         } catch (NumberFormatException e) {
449             LOG.trace(DEFAULT_LOG_MSG, hexString, e);
450             // Hex failed to parse, just return the existing string
451             return hexString;
452         }
453         return sb.toString();
454     }

530     /**
531      * Attempts to parse a string of the form [DD-[hh:]]mm:ss[.ddd] to a number of
532      * milliseconds. If it fails, returns the default.
533      *
534      * @param s
535      *            The string to parse
536      * @param defaultLong
537      *            The value to return if parsing fails
538      * @return The parsed number of seconds, or the default if parsing fails
539      */
540     public static long parseDHMSOrDefault(String s, long defaultLong) {
541         Matcher m = DHMS.matcher(s);
542         if (m.matches()) {
543             long milliseconds = 0L;
544             if (m.group(1) != null) {
545                 milliseconds += parseLongOrDefault(m.group(1), 0L) * 86400000L;
546             }
547             if (m.group(2) != null) {
548                 milliseconds += parseLongOrDefault(m.group(2), 0L) * 3600000L;
549             }
550             if (m.group(3) != null) {
551                 milliseconds += parseLongOrDefault(m.group(3), 0L) * 60000L;
552             }
553             milliseconds += parseLongOrDefault(m.group(4), 0L) * 1000L;
554             milliseconds += (long) (1000 * parseDoubleOrDefault("0." + m.group(5), 0d));
555             return milliseconds;
556         }
557         return defaultLong;
558     }

653     /**
654      * Removes all matching sub strings from the string. More efficient than regexp.
655      *
656      * @param original
657      *            source String to remove from
658      * @param toRemove
659      *            the sub string to be removed
660      * @return The string with all matching substrings removed
661      */
662     public static String removeMatchingString(final String original, final String toRemove) {
663         if (original == null || original.isEmpty() || toRemove == null || toRemove.isEmpty()) {
664             return original;
665         }
666 
667         int matchIndex = original.indexOf(toRemove, 0);
668         if (matchIndex == -1) {
669             return original;
670         }
671 
672         StringBuilder buffer = new StringBuilder(original.length() - toRemove.length());
673         int currIndex = 0;
674         do {
675             buffer.append(original.substring(currIndex, matchIndex));
676             currIndex = matchIndex + toRemove.length();
677             matchIndex = original.indexOf(toRemove, currIndex);
678         } while (matchIndex != -1);
679

684     /**
685      * Parses a delimited string to an array of longs. Optimized for processing
686      * predictable-length arrays such as outputs of reliably formatted Linux proc or
687      * sys filesystem, minimizing new object creation. Users should perform other
688      * sanity checks of data.
689      *
690      * As a special case, non-numeric fields (such as UUIDs in OpenVZ) at the end of
691      * the list are ignored. Values greater than the max long value return the max
692      * long value.
693      *
694      * The indices parameters are referenced assuming the length as specified, and
695      * leading characters are ignored. For example, if the string is "foo 12 34 5"
696      * and the length is 3, then index 0 is 12, index 1 is 34, and index 2 is 5.
697      *
698      * @param s
699      *            The string to parse
700      * @param indices
701      *            An array indicating which indexes should be populated in the final
702      *            array; other values will be skipped. This idex is zero-referenced
703      *            assuming the rightmost delimited fields of the string contain the
704      *            array.
705      * @param length
706      *            The total number of elements in the string array. It is
707      *            permissible for the string to have more elements than this;
708      *            leading elements will be ignored. This should be calculated once
709      *            per text format by {@link #countStringToLongArray}.
710      * @param delimiter
711      *            The character to delimit by.
712      * @return If successful, an array of parsed longs. If parsing errors occurred,
713      *         will be an array of zeros.
714      */
715     public static long[] parseStringToLongArray(String s, int[] indices, int length, char delimiter) {
716         long[] parsed = new long[indices.length];
717         // Iterate from right-to-left of String
718         // Fill right to left of result array using index array
719         int charIndex = s.length();
720         int parsedIndex = indices.length - 1;
721         int stringIndex = length - 1;
722 
723         int power = 0;
724         int c;
725         boolean delimCurrent = false;
726         boolean numeric = true;
727         boolean numberFound = false; // ignore nonnumeric at end
728         boolean dashSeen = false; // to flag uuids as nonnumeric
729         while (--charIndex > 0 && parsedIndex >= 0) {
730             c = s.charAt(charIndex);
731             if (c == delimiter) {
732                 // first parseable number?
733                 if (!numberFound && numeric) {
734                     numberFound = true;
735                 }
736                 if (!delimCurrent) {
737                     if (numberFound && indices[parsedIndex] == stringIndex--) {
738                         parsedIndex--;
739                     }
740                     delimCurrent = true;
741                     power = 0;
742                     dashSeen = false;
743                     numeric = true;
744                 }
745             } else if (indices[parsedIndex] != stringIndex || c == '+' || !numeric) {
746                 // Doesn't impact parsing, ignore
747                 delimCurrent = false;
748             } else if (c >= '0' && c <= '9' && !dashSeen) {
749                 if (power > 18 || power == 17 && c == '9' && parsed[parsedIndex] > 223372036854775807L) {
750                     parsed[parsedIndex] = Long.MAX_VALUE;
751                 } else {
752                     parsed[parsedIndex] += (c - '0') * ParseUtil.POWERS_OF_TEN[power++];
753                 }
754                 delimCurrent = false;
755             } else if (c == '-') {
756                 parsed[parsedIndex] *= -1L;
757                 delimCurrent = false;
758                 dashSeen = true;
759             } else {
760                 // Flag as nonnumeric and continue unless we've seen a numeric
761                 // error on everything else
762                 if (numberFound) {
763                     LOG.error("Illegal character parsing string '{}' to long array: {}", s, s.charAt(charIndex));
764                     return new long[indices.length];
765                 }
766                 parsed[parsedIndex] = 0;
767                 numeric = false;
768             }
769         }
770         if (parsedIndex > 0) {
771             LOG.error("Not enough fields in string '{}' parsing to long array: {}", s, indices.length - parsedIndex);
772             return new long[indices.length];
773         }
774         return parsed;
775     }

777     /**
778      * Parses a delimited string to count elements of an array of longs. Intended to
779      * be called once to calculate the {@code length} field for
780      * {@link #parseStringToLongArray}.
781      *
782      * As a special case, non-numeric fields (such as UUIDs in OpenVZ) at the end of
783      * the list are ignored.
784      *
785      * @param s
786      *            The string to parse
787      * @param delimiter
788      *            The character to delimit by
789      * @return The number of parsable long values which follow the last unparsable
790      *         value.
791      */
792     public static int countStringToLongArray(String s, char delimiter) {
793         // Iterate from right-to-left of String
794         // Fill right to left of result array using index array
795         int charIndex = s.length();
796         int numbers = 0;
797 
798         int c;
799         boolean delimCurrent = false;
800         boolean numeric = true;
801         boolean dashSeen = false; // to flag uuids as nonnumeric
802         while (--charIndex > 0) {
803             c = s.charAt(charIndex);
804             if (c == delimiter) {
805                 if (!delimCurrent) {
806                     if (numeric) {
807                         numbers++;
808                     }
809                     delimCurrent = true;
810                     dashSeen = false;
811                     numeric = true;
812                 }
813             } else if (c == '+' || !numeric) {
814                 // Doesn't impact parsing, ignore
815                 delimCurrent = false;
816             } else if (c >= '0' && c <= '9' && !dashSeen) {
817                 delimCurrent = false;
818             } else if (c == '-') {
819                 delimCurrent = false;
820                 dashSeen = true;
821             } else {
822                 // we found non-digit or delimiter. If not last field, exit
823                 if (numbers > 0) {
824                     return numbers;
825                 }
826                 // Else flag as nonnumeric and continue
827                 numeric = false;
828             }
829         }
830         // We got to beginning of string with only numbers, count start as a delimiter
831         // and exit
832         return numbers + 1;
833     }

962             case 'K':
963             case 'k':

1084      * Parse an integer array to an IPv4 or IPv6 as appropriate.
1085      * <p>
1086      * Intended for use on Utmp structures's {@code ut_addr_v6} element.
1087      *
1088      * @param utAddrV6
1089      *            An array of 4 integers representing an IPv6 address. IPv4 address
1090      *            uses just utAddrV6[0]
1091      * @return A string representation of the IP address.
1092      */
1093     public static String parseUtAddrV6toIP(int[] utAddrV6) {
1094         if (utAddrV6.length != 4) {
1095             throw new IllegalArgumentException("ut_addr_v6 must have exactly 4 elements");
1096         }
1097         // IPv4 has only first element
1098         if (utAddrV6[1] == 0 && utAddrV6[2] == 0 && utAddrV6[3] == 0) {
1099             // Special case for all 0's
1100             if (utAddrV6[0] == 0) {
1101                 return "::";
1102             }
1103             // Parse using InetAddress
1104             byte[] ipv4 = ByteBuffer.allocate(4).putInt(utAddrV6[0]).array();
1105             try {
1106                 return InetAddress.getByAddress(ipv4).getHostAddress();
1107             } catch (UnknownHostException e) {
1108                 // Shouldn't happen with length 4 or 16
1109                 return Constants.UNKNOWN;
1110             }
1111         }
1112         // Parse all 16 bytes
1113         byte[] ipv6 = ByteBuffer.allocate(16).putInt(utAddrV6[0]).putInt(utAddrV6[1]).putInt(utAddrV6[2])
1114                 .putInt(utAddrV6[3]).array();
1115         try {
1116             return InetAddress.getByAddress(ipv6).getHostAddress()
1117                     .replaceAll("((?:(?:^|:)0+\\b){2,}):?(?!\\S*\\b\\1:0+\\b)(\\S*)", "::$2");
1118         } catch (UnknownHostException e) {
1119             // Shouldn't happen with length 4 or 16
1120             return Constants.UNKNOWN;
1121         }
1122     }
