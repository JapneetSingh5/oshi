213     /**
214      * Get System-wide CPU Load tick counters. Returns an array with seven elements
215      * representing milliseconds spent in User (0), Nice (1), System (2), Idle (3),
216      * IOwait (4), Hardware interrupts (IRQ) (5), Software interrupts/DPC (SoftIRQ)
217      * (6), or Steal (7) states. Use
218      * {@link oshi.hardware.CentralProcessor.TickType#getIndex()} to retrieve the

221      * <p>
222      * Note that while tick counters are in units of milliseconds, they may advance
223      * in larger increments along with (platform dependent) clock ticks. For
224      * example, by default Windows clock ticks are 1/64 of a second (about 15 or 16

226      * but usually 1/100 of a second (10 milliseconds).

229      * information is not available on macOS, so these ticks will always be zero.
230      * <p>

232      * IOWait ticks. Similarly, IRQ, SoftIRQ, and Steal ticks should be added to the
233      * System value to get the total. System ticks also include time executing other
234      * virtual hosts (steal).
235      *
236      * @return An array of 7 long values representing time spent in User, Nice,
237      *         System, Idle, IOwait, IRQ, SoftIRQ, and Steal states.

343     /**
344      * Index of CPU tick counters in the {@link #getSystemCpuLoadTicks()} and
345      * {@link #getProcessorCpuLoadTicks()} arrays.
346      */
347     enum TickType {
348         /**
349          * CPU utilization that occurred while executing at the user level
350          * (application).
351          */
352         USER(0),
353         /**
354          * CPU utilization that occurred while executing at the user level with nice
355          * priority.
356          */
357         NICE(1),
358         /**
359          * CPU utilization that occurred while executing at the system level (kernel).
360          */
361         SYSTEM(2),
362         /**
363          * Time that the CPU or CPUs were idle and the system did not have an
364          * outstanding disk I/O request.
365          */
366         IDLE(3),
367         /**
368          * Time that the CPU or CPUs were idle during which the system had an
369          * outstanding disk I/O request.
370          */
371         IOWAIT(4),
372         /**
373          * Time that the CPU used to service hardware IRQs
374          */
375         IRQ(5),
376         /**
377          * Time that the CPU used to service soft IRQs
378          */
379         SOFTIRQ(6),
380         /**
381          * Time which the hypervisor dedicated for other guests in the system. Only
382          * supported on Linux.
383          */
384         STEAL(7);

521         private static final String OSHI_ARCHITECTURE_PROPERTIES = "oshi.architecture.properties";
522 
523         // Provided in constructor
524         private final String cpuVendor;
525         private final String cpuName;
526         private final String cpuFamily;
527         private final String cpuModel;
528         private final String cpuStepping;
529         private final String processorID;
530         private final String cpuIdentifier;
531         private final boolean cpu64bit;
532         private final long cpuVendorFreq;

534         private final Supplier<String> microArchictecture = memoize(this::queryMicroarchitecture);

536         public ProcessorIdentifier(String cpuVendor, String cpuName, String cpuFamily, String cpuModel,
537                 String cpuStepping, String processorID, boolean cpu64bit) {
538             this.cpuVendor = cpuVendor;
539             this.cpuName = cpuName;
540             this.cpuFamily = cpuFamily;
541             this.cpuModel = cpuModel;
542             this.cpuStepping = cpuStepping;
543             this.processorID = processorID;
544             this.cpu64bit = cpu64bit;
545 
546             // Build Identifier
547             StringBuilder sb = new StringBuilder();
548             if (cpuVendor.contentEquals("GenuineIntel")) {
549                 sb.append(cpu64bit ? "Intel64" : "x86");
550             } else {
551                 sb.append(cpuVendor);
552             }
553             sb.append(" Family ").append(cpuFamily);
554             sb.append(" Model ").append(cpuModel);
555             sb.append(" Stepping ").append(cpuStepping);
556             this.cpuIdentifier = sb.toString();
557 
558             // Parse Freq from name string
559             Pattern pattern = Pattern.compile("@ (.*)$");
560             Matcher matcher = pattern.matcher(cpuName);
561             if (matcher.find()) {
562                 String unit = matcher.group(1);
563                 this.cpuVendorFreq = ParseUtil.parseHertz(unit);
564             } else {
565                 this.cpuVendorFreq = -1L;
566             }
567         }

681         private String queryMicroarchitecture() {
682             Properties archProps = FileUtil.readPropertiesFromFilename(OSHI_ARCHITECTURE_PROPERTIES);
683             // Intel is default, no prefix
684             StringBuilder sb = new StringBuilder();
685             // AMD and ARM properties have prefix
686             if (this.cpuVendor.contains("AMD")) {
687                 sb.append("amd.");
688             } else if (this.getVendor().contains("ARM")) {
689                 sb.append("arm.");
690             }
691             sb.append(this.cpuFamily);
692             // Check for match with only family
693             String arch = archProps.getProperty(sb.toString());
694 
695             if (Util.isBlank(arch)) {
696                 // Append model
697                 sb.append('.').append(this.cpuModel);
698             }
699             arch = archProps.getProperty(sb.toString());
700 
701             if (Util.isBlank(arch)) {
702                 // Append stepping
703                 sb.append('.').append(this.cpuStepping);
704             }
705             arch = archProps.getProperty(sb.toString());
706 
707             return Util.isBlank(arch) ? Constants.UNKNOWN : arch;
708         }
