54      @Override
55      protected ProcessorIdentifier queryProcessorId() {
56          String cpuVendor = SysctlUtil.sysctl("machdep.cpu.vendor", "");
57          String cpuName = SysctlUtil.sysctl("machdep.cpu.brand_string", "");
58          int i = SysctlUtil.sysctl("machdep.cpu.stepping", -1);
59          String cpuStepping = i < 0 ? "" : Integer.toString(i);
60          i = SysctlUtil.sysctl("machdep.cpu.model", -1);
61          String cpuModel = i < 0 ? "" : Integer.toString(i);
62          i = SysctlUtil.sysctl("machdep.cpu.family", -1);
63          String cpuFamily = i < 0 ? "" : Integer.toString(i);
64          long processorIdBits = 0L;
65          processorIdBits |= SysctlUtil.sysctl("machdep.cpu.signature", 0);
66          processorIdBits |= (SysctlUtil.sysctl("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
67          String processorID = String.format("%016X", processorIdBits);
68          boolean cpu64bit = SysctlUtil.sysctl("hw.cpu64bit_capable", 0) != 0;
69  
70          return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit);
71      }
72  
73      @Override
74      protected List<LogicalProcessor> initProcessorCounts() {
75          int logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
76          int physicalProcessorCount = SysctlUtil.sysctl("hw.physicalcpu", 1);
77          int physicalPackageCount = SysctlUtil.sysctl("hw.packages", 1);
78          List<LogicalProcessor> logProcs = new ArrayList<>(logicalProcessorCount);
79          for (int i = 0; i < logicalProcessorCount; i++) {
80              logProcs.add(new LogicalProcessor(i, i * physicalProcessorCount / logicalProcessorCount,
81                      i * physicalPackageCount / logicalProcessorCount));
82          }
83          return logProcs;
84      }

86      @Override
87      public long[] querySystemCpuLoadTicks() {
88          long[] ticks = new long[TickType.values().length];
89          int machPort = SystemB.INSTANCE.mach_host_self();
90          HostCpuLoadInfo cpuLoadInfo = new HostCpuLoadInfo();
91          if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo,
92                  new IntByReference(cpuLoadInfo.size()))) {
93              LOG.error("Failed to get System CPU ticks. Error code: {} ", Native.getLastError());
94              return ticks;
95          }
96  
97          ticks[TickType.USER.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
98          ticks[TickType.NICE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
99          ticks[TickType.SYSTEM.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
100         ticks[TickType.IDLE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
101         // Leave IOWait and IRQ values as 0
102         return ticks;
103     }

117     @Override
118     public double[] getSystemLoadAverage(int nelem) {
119         if (nelem < 1 || nelem > 3) {
120             throw new IllegalArgumentException("Must include from one to three elements.");
121         }
122         double[] average = new double[nelem];
123         int retval = SystemB.INSTANCE.getloadavg(average, nelem);
124         if (retval < nelem) {
125             Arrays.fill(average, -1d);
126         }
127         return average;
128     }

130     @Override
131     public long[][] queryProcessorCpuLoadTicks() {
132         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
133 
134         int machPort = SystemB.INSTANCE.mach_host_self();
135 
136         IntByReference procCount = new IntByReference();
137         PointerByReference procCpuLoadInfo = new PointerByReference();
138         IntByReference procInfoCount = new IntByReference();
139         if (0 != SystemB.INSTANCE.host_processor_info(machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
140                 procCpuLoadInfo, procInfoCount)) {
141             LOG.error("Failed to update CPU Load. Error code: {}", Native.getLastError());
142             return ticks;
143         }
144 
145         int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray(0, procInfoCount.getValue());
146         for (int cpu = 0; cpu < procCount.getValue(); cpu++) {
147             int offset = cpu * SystemB.CPU_STATE_MAX;
148             ticks[cpu][TickType.USER.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_USER]);
149             ticks[cpu][TickType.NICE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_NICE]);
150             ticks[cpu][TickType.SYSTEM.getIndex()] = FormatUtil
151                     .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_SYSTEM]);
152             ticks[cpu][TickType.IDLE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_IDLE]);
153         }
154         return ticks;
155     }

157     @Override
158     public long queryContextSwitches() {
159         int machPort = SystemB.INSTANCE.mach_host_self();
160         VMMeter vmstats = new VMMeter();
161         if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_VM_INFO, vmstats,
162                 new IntByReference(vmstats.size()))) {
163             LOG.error("Failed to update vmstats. Error code: {}", Native.getLastError());
164             return -1;
165         }
166         return ParseUtil.unsignedIntToLong(vmstats.v_swtch);
167     }
