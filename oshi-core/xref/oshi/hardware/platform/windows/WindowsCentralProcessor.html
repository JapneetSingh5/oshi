87      @Override
88      protected ProcessorIdentifier queryProcessorId() {
89          String cpuVendor = "";
90          String cpuName = "";
91          String cpuIdentifier = "";
92          String cpuFamily = "";
93          String cpuModel = "";
94          String cpuStepping = "";
95          String processorID;
96          boolean cpu64bit = false;
97  
98          final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
99          String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
100         if (processorIds.length > 0) {
101             String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
102             cpuVendor = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
103                     "VendorIdentifier");
104             cpuName = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
105                     "ProcessorNameString");
106             cpuIdentifier = Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
107                     "Identifier");
108         }
109         if (!cpuIdentifier.isEmpty()) {
110             cpuFamily = parseIdentifier(cpuIdentifier, "Family");
111             cpuModel = parseIdentifier(cpuIdentifier, "Model");
112             cpuStepping = parseIdentifier(cpuIdentifier, "Stepping");
113         }
114         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
115         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
116         int processorArchitecture = sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue();
117         if (processorArchitecture == 9 // PROCESSOR_ARCHITECTURE_AMD64
118                 || processorArchitecture == 12 // PROCESSOR_ARCHITECTURE_ARM64
119                 || processorArchitecture == 6) { // PROCESSOR_ARCHITECTURE_IA64
120             cpu64bit = true;
121         }
122         WmiResult<ProcessorIdProperty> processorId = Win32Processor.queryProcessorId();
123         if (processorId.getResultCount() > 0) {
124             processorID = WmiUtil.getString(processorId, ProcessorIdProperty.PROCESSORID, 0);
125         } else {
126             processorID = createProcessorID(cpuStepping, cpuModel, cpuFamily,
127                     cpu64bit ? new String[] { "ia64" } : new String[0]);
128         }
129         return new ProcessorIdentifier(cpuVendor, cpuName, cpuFamily, cpuModel, cpuStepping, processorID, cpu64bit);
130     }

132     /**
133      * Parses identifier string
134      *
135      * @param identifier
136      *            the full identifier string
137      * @param key
138      *            the key to retrieve
139      * @return the string following id
140      */
141     private static String parseIdentifier(String identifier, String key) {
142         String[] idSplit = ParseUtil.whitespaces.split(identifier);
143         boolean found = false;
144         for (String s : idSplit) {
145             // If key string found, return next value
146             if (found) {
147                 return s;
148             }
149             found = s.equals(key);
150         }
151         // If key string not found, return empty string
152         return "";
153     }

164     private List<LogicalProcessor> getLogicalProcessorInformationEx() {
165         // Collect a list of logical processors on each physical core and
166         // package. These will be 64-bit bitmasks.
167         SYSTEM_LOGICAL_PROCESSOR_INFORMATION_EX[] procInfo = Kernel32Util
168                 .getLogicalProcessorInformationEx(WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationAll);
169         List<GROUP_AFFINITY[]> packages = new ArrayList<>();
170         List<NUMA_NODE_RELATIONSHIP> numaNodes = new ArrayList<>();
171         List<GROUP_AFFINITY> cores = new ArrayList<>();
172 
173         for (int i = 0; i < procInfo.length; i++) {
174             switch (procInfo[i].relationship) {
175             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage:
176                 packages.add(((PROCESSOR_RELATIONSHIP) procInfo[i]).groupMask);
177                 break;
178             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationNumaNode:
179                 numaNodes.add((NUMA_NODE_RELATIONSHIP) procInfo[i]);
180                 break;
181             case LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore:
182                 cores.add(((PROCESSOR_RELATIONSHIP) procInfo[i]).groupMask[0]);
183                 break;
184             default:
185                 // Ignore Group and Cache info
186                 break;
187             }
188         }
189         // Windows doesn't define core and package numbers, so we sort the lists
190         // so core and package numbers increment consistently with processor
191         // numbers/bitmasks, ordered in groups
192         cores.sort(Comparator.comparing(c -> c.group * 64L + c.mask.longValue()));
193         packages.sort(Comparator.comparing(p -> p[0].group * 64L + p[0].mask.longValue()));
194 
195         // Iterate Logical Processors and use bitmasks to match packages, cores,
196         // and NUMA nodes
197         List<LogicalProcessor> logProcs = new ArrayList<>();
198         for (GROUP_AFFINITY coreMask : cores) {
199             int group = coreMask.group;
200             long mask = coreMask.mask.longValue();
201             // Iterate mask for logical processor numbers
202             int lowBit = Long.numberOfTrailingZeros(mask);
203             int hiBit = 63 - Long.numberOfLeadingZeros(mask);
204             for (int lp = lowBit; lp <= hiBit; lp++) {
205                 if ((mask & (1L << lp)) > 0) {
206                     LogicalProcessor logProc = new LogicalProcessor(lp, getMatchingCore(cores, group, lp),
207                             getMatchingPackage(packages, group, lp), getMatchingNumaNode(numaNodes, group, lp), group);
208                     logProcs.add(logProc);
209                 }
210             }
211         }
212         // Sort by numaNode and then logical processor number to match
213         // PerfCounter/WMI ordering
214         logProcs.sort(Comparator.comparing(LogicalProcessor::getNumaNode)
215                 .thenComparing(LogicalProcessor::getProcessorNumber));
216         // Save numaNode,Processor lookup for future PerfCounter instance lookup
217         int lp = 0;
218         this.numaNodeProcToLogicalProcMap = new HashMap<>();
219         for (LogicalProcessor logProc : logProcs) {
220             numaNodeProcToLogicalProcMap
221                     .put(String.format("%d,%d", logProc.getNumaNode(), logProc.getProcessorNumber()), lp++);
222         }
223         return logProcs;
224     }

226     private static int getMatchingPackage(List<GROUP_AFFINITY[]> packages, int g, int lp) {
227         for (int i = 0; i < packages.size(); i++) {
228             for (int j = 0; j < packages.get(i).length; j++) {
229                 if ((packages.get(i)[j].mask.longValue() & (1L << lp)) > 0 && packages.get(i)[j].group == g) {
230                     return i;
231                 }
232             }
233         }
234         return 0;
235     }
236 
237     private static int getMatchingNumaNode(List<NUMA_NODE_RELATIONSHIP> numaNodes, int g, int lp) {
238         for (int j = 0; j < numaNodes.size(); j++) {
239             if ((numaNodes.get(j).groupMask.mask.longValue() & (1L << lp)) > 0
240                     && numaNodes.get(j).groupMask.group == g) {
241                 return numaNodes.get(j).nodeNumber;
242             }
243         }
244         return 0;
245     }
246 
247     private static int getMatchingCore(List<GROUP_AFFINITY> cores, int g, int lp) {
248         for (int j = 0; j < cores.size(); j++) {
249             if ((cores.get(j).mask.longValue() & (1L << lp)) > 0 && cores.get(j).group == g) {
250                 return j;
251             }
252         }
253         return 0;
254     }

256     private List<LogicalProcessor> getLogicalProcessorInformation() {
257         // Collect a list of logical processors on each physical core and
258         // package. These will be 64-bit bitmasks.
259         List<Long> packageMaskList = new ArrayList<>();
260         List<Long> coreMaskList = new ArrayList<>();
261         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
262         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
263             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage) {
264                 packageMaskList.add(proc.processorMask.longValue());
265             } else if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
266                 coreMaskList.add(proc.processorMask.longValue());
267             }
268         }
269         // Sort the list (natural ordering) so core and package numbers
270         // increment as expected.
271         coreMaskList.sort(null);
272         packageMaskList.sort(null);
273 
274         // Assign logical processors to cores and packages
275         List<LogicalProcessor> logProcs = new ArrayList<>();
276         for (int core = 0; core < coreMaskList.size(); core++) {
277             long coreMask = coreMaskList.get(core);
278             // Lowest and Highest set bits, indexing from 0
279             int lowBit = Long.numberOfTrailingZeros(coreMask);
280             int hiBit = 63 - Long.numberOfLeadingZeros(coreMask);
281             // Create logical processors for this core
282             for (int i = lowBit; i <= hiBit; i++) {
283                 if ((coreMask & (1L << i)) > 0) {
284                     LogicalProcessor logProc = new LogicalProcessor(i, core,
285                             getBitMatchingPackageNumber(packageMaskList, i));
286                     logProcs.add(logProc);
287                 }
288             }
289         }
290         return logProcs;
291     }

293     /**
294      * Iterate over the package mask list and find a matching mask index
295      *
296      * @param packageMaskList
297      *            The list of bitmasks to iterate
298      * @param logProc
299      *            The bit to find matching mask
300      * @return The index of the list which matched the bit
301      */
302     private static int getBitMatchingPackageNumber(List<Long> packageMaskList, int logProc) {
303         for (int i = 0; i < packageMaskList.size(); i++) {
304             if ((packageMaskList.get(i).longValue() & (1L << logProc)) > 0) {
305                 return i;
306             }
307         }
308         return 0;
309     }

311     @Override
312     public long[] querySystemCpuLoadTicks() {
313         long[] ticks = new long[TickType.values().length];
314         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
315         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
316         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
317         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
318             LOG.error("Failed to update system idle/kernel/user times. Error code: {}", Native.getLastError());
319             return ticks;
320         }
321         // IOwait:
322         // Windows does not measure IOWait.
323 
324         // IRQ and ticks:
325         // Percent time raw value is cumulative 100NS-ticks
326         // Divide by 10_000 to get milliseconds
327 
328         Map<SystemTickCountProperty, Long> valueMap = ProcessorInformation.querySystemCounters();
329         ticks[TickType.IRQ.getIndex()] = valueMap.getOrDefault(SystemTickCountProperty.PERCENTINTERRUPTTIME, 0L)
330                 / 10_000L;
331         ticks[TickType.SOFTIRQ.getIndex()] = valueMap.getOrDefault(SystemTickCountProperty.PERCENTDPCTIME, 0L)
332                 / 10_000L;
333 
334         ticks[TickType.IDLE.getIndex()] = lpIdleTime.toDWordLong().longValue() / 10_000L;
335         ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.toDWordLong().longValue() / 10_000L
336                 - ticks[TickType.IDLE.getIndex()];
337         ticks[TickType.USER.getIndex()] = lpUserTime.toDWordLong().longValue() / 10_000L;
338         // Additional decrement to avoid double counting in the total array
339         ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()] + ticks[TickType.SOFTIRQ.getIndex()];
340         return ticks;
341     }

343     @Override
344     public long[] queryCurrentFreq() {
345         if (VersionHelpers.IsWindows7OrGreater()) {
346             Pair<List<String>, Map<ProcessorFrequencyProperty, List<Long>>> instanceValuePair = ProcessorInformation
347                     .queryFrequencyCounters();
348             List<String> instances = instanceValuePair.getA();
349             Map<ProcessorFrequencyProperty, List<Long>> valueMap = instanceValuePair.getB();
350             List<Long> percentMaxList = valueMap.get(ProcessorFrequencyProperty.PERCENTOFMAXIMUMFREQUENCY);
351             if (!instances.isEmpty()) {
352                 long maxFreq = this.getMaxFreq();
353                 long[] freqs = new long[getLogicalProcessorCount()];
354                 for (int p = 0; p < instances.size(); p++) {
355                     int cpu = instances.get(p).contains(",")
356                             ? numaNodeProcToLogicalProcMap.getOrDefault(instances.get(p), 0)
357                             : ParseUtil.parseIntOrDefault(instances.get(p), 0);
358                     if (cpu >= getLogicalProcessorCount()) {
359                         continue;
360                     }
361                     freqs[cpu] = percentMaxList.get(cpu) * maxFreq / 100L;
362                 }
363                 return freqs;
364             }
365         }
366         // If <Win7 or anything failed in PDH/WMI, use the native call
367         return queryNTPower(2); // Current is field index 2
368     }

370     @Override
371     public long queryMaxFreq() {
372         long[] freqs = queryNTPower(1); // Max is field index 1
373         return Arrays.stream(freqs).max().getAsLong();
374     }

376     /**
377      * Call CallNTPowerInformation for Processor information and return an array of
378      * the specified index
379      *
380      * @param fieldIndex
381      *            The field, in order as defined in the
382      *            {@link PowrProf#PROCESSOR_INFORMATION} structure.
383      * @return The array of values.
384      */
385     private long[] queryNTPower(int fieldIndex) {
386         ProcessorPowerInformation ppi = new ProcessorPowerInformation();
387         long[] freqs = new long[getLogicalProcessorCount()];
388         int bufferSize = ppi.size() * freqs.length;
389         Memory mem = new Memory(bufferSize);
390         if (0 != PowrProf.INSTANCE.CallNtPowerInformation(POWER_INFORMATION_LEVEL.ProcessorInformation, null, 0, mem,
391                 bufferSize)) {
392             LOG.error("Unable to get Processor Information");
393             Arrays.fill(freqs, -1L);
394             return freqs;
395         }
396         for (int i = 0; i < freqs.length; i++) {
397             ppi = new ProcessorPowerInformation(mem.share(i * (long) ppi.size()));
398             if (fieldIndex == 1) { // Max
399                 freqs[i] = ppi.maxMhz * 1_000_000L;
400             } else if (fieldIndex == 2) { // Current
401                 freqs[i] = ppi.currentMhz * 1_000_000L;
402             } else {
403                 freqs[i] = -1L;
404             }
405         }
406         return freqs;
407     }

409     @Override
410     public double[] getSystemLoadAverage(int nelem) {
411         if (nelem < 1 || nelem > 3) {
412             throw new IllegalArgumentException("Must include from one to three elements.");
413         }
414         double[] average = new double[nelem];
415         // Windows doesn't have load average
416         for (int i = 0; i < average.length; i++) {
417             average[i] = -1;
418         }
419         return average;
420     }

422     @Override
423     public long[][] queryProcessorCpuLoadTicks() {
424         Pair<List<String>, Map<ProcessorTickCountProperty, List<Long>>> instanceValuePair = ProcessorInformation
425                 .queryProcessorCounters();
426         List<String> instances = instanceValuePair.getA();
427         Map<ProcessorTickCountProperty, List<Long>> valueMap = instanceValuePair.getB();
428         List<Long> systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
429         List<Long> userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
430         List<Long> irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
431         List<Long> softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
432         // % Processor Time is actually Idle time
433         List<Long> idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
434 
435         long[][] ticks = new long[getLogicalProcessorCount()][TickType.values().length];
436         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
437                 || idleList == null) {
438             return ticks;
439         }
440         for (int p = 0; p < instances.size(); p++) {
441             int cpu = instances.get(p).contains(",") ? numaNodeProcToLogicalProcMap.getOrDefault(instances.get(p), 0)
442                     : ParseUtil.parseIntOrDefault(instances.get(p), 0);
443             if (cpu >= getLogicalProcessorCount()) {
444                 continue;
445             }
446             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(cpu);
447             ticks[cpu][TickType.USER.getIndex()] = userList.get(cpu);
448             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(cpu);
449             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(cpu);
450             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(cpu);
451 
452             // Additional decrement to avoid double counting in the
453             // total array
454             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
455                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
456 
457             // Raw value is cumulative 100NS-ticks
458             // Divide by 10_000 to get milliseconds
459             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10_000L;
460             ticks[cpu][TickType.USER.getIndex()] /= 10_000L;
461             ticks[cpu][TickType.IRQ.getIndex()] /= 10_000L;
462             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10_000L;
463             ticks[cpu][TickType.IDLE.getIndex()] /= 10_000L;
464         }
465         // Skipping nice and IOWait, they'll stay 0
466         return ticks;
467     }
