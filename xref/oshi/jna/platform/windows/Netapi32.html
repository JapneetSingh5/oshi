36  public interface Netapi32 extends com.sun.jna.platform.win32.Netapi32 {
37  
38      Netapi32 INSTANCE = Native.load("Netapi32", Netapi32.class, W32APIOptions.DEFAULT_OPTIONS);
39  
40      int MAX_PREFERRED_LENGTH = -1;
41  
42      /**
43       * Contains information about the session, including name of the computer; name
44       * of the user; and active and idle times for the session.
45       */
46      @FieldOrder({ "sesi10_cname", "sesi10_username", "sesi10_time", "sesi10_idle_time" })
47      class SESSION_INFO_10 extends Structure {
48          public String sesi10_cname;
49          public String sesi10_username;
50          public int sesi10_time;
51          public int sesi10_idle_time;
52  
53          public SESSION_INFO_10() {
54              super(W32APITypeMapper.UNICODE);
55          }
56  
57          public SESSION_INFO_10(Pointer p) {
58              super(p, Structure.ALIGN_DEFAULT, W32APITypeMapper.UNICODE);
59              read();
60          }
61      }
62  
63      /**
64       * Provides information about sessions established on a server.
65       *
66       * @param servername
67       *            Pointer to a string that specifies the DNS or NetBIOS name of the
68       *            remote server on which the function is to execute. If this
69       *            parameter is NULL, the local computer is used.
70       * @param UncClientName
71       *            Pointer to a string that specifies the name of the computer
72       *            session for which information is to be returned. If this parameter
73       *            is NULL, NetSessionEnum returns information for all computer
74       *            sessions on the server.
75       * @param username
76       *            Pointer to a string that specifies the name of the user for which
77       *            information is to be returned. If this parameter is NULL,
78       *            NetSessionEnum returns information for all users.
79       * @param level
80       *            Specifies the information level of the data. This parameter can be
81       *            one of 0, 1, 2, 10, 502.
82       * @param bufptr
83       *            Pointer to the buffer that receives the data. The format of this
84       *            data depends on the value of the level parameter, for example
85       *            {@code SESSION_INFO_0} for level 0.
86       *            <p>
87       *            This buffer is allocated by the system and must be freed using the
88       *            {@link #NetApiBufferFree} function. Note that you must free the
89       *            buffer even if the function fails with {@code ERROR_MORE_DATA}.
90       * @param prefmaxlen
91       *            Specifies the preferred maximum length of returned data, in bytes.
92       *            If you specify {@link #MAX_PREFERRED_LENGTH}, the function
93       *            allocates the amount of memory required for the data. If you
94       *            specify another value in this parameter, it can restrict the
95       *            number of bytes that the function returns. If the buffer size is
96       *            insufficient to hold all entries, the function returns
97       *            {@code ERROR_MORE_DATA}.
98       * @param entriesread
99       *            Pointer to a value that receives the count of elements actually
100      *            enumerated.
101      * @param totalentries
102      *            Pointer to a value that receives the total number of entries that
103      *            could have been enumerated from the current resume position. Note
104      *            that applications should consider this value only as a hint.
105      * @param resume_handle
106      *            Pointer to a value that contains a resume handle which is used to
107      *            continue an existing session search. The handle should be zero on
108      *            the first call and left unchanged for subsequent calls. If
109      *            resume_handle is NULL, no resume handle is stored.
110      * @return If the function succeeds, the return value is NERR_Success (0). If
111      *         the function fails, the return value is an error code.
112      */
113     int NetSessionEnum(WString servername, WString UncClientName, WString username, int level,
114             PointerByReference bufptr, int prefmaxlen, IntByReference entriesread, IntByReference totalentries,
115             IntByReference resume_handle);
116 }
