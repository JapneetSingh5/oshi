99      private static final long MAX_WINDOWS_HANDLES;
100     static {
101         // Determine whether 32-bit or 64-bit handle limit, although both are
102         // essentially infinite for practical purposes. See
103         // https://blogs.technet.microsoft.com/markrussinovich/2009/09/29/pushing-the-limits-of-windows-handles/
104         if (System.getenv("ProgramFiles(x86)") == null) {
105             MAX_WINDOWS_HANDLES = 16_777_216L - 32_768L;
106         } else {
107             MAX_WINDOWS_HANDLES = 16_777_216L - 65_536L;
108         }

121     @Override
122     public OSFileStore[] getFileStores(boolean localOnly) {
123         // Create list to hold results
124         ArrayList<OSFileStore> result;
125 
126         // Begin with all the local volumes
127         result = getLocalVolumes(null);
128 
129         // Build a map of existing mount point to OSFileStore
130         Map<String, OSFileStore> volumeMap = new HashMap<>();
131         for (OSFileStore volume : result) {
132             volumeMap.put(volume.getMount(), volume);
133         }
134 
135         // Iterate through volumes in WMI and update description (if it exists)
136         // or add new if it doesn't (expected for network drives)
137         for (OSFileStore wmiVolume : getWmiVolumes(null, localOnly)) {
138             if (volumeMap.containsKey(wmiVolume.getMount())) {
139                 // If the volume is already in our list, update the name field
140                 // using WMI's more verbose name and update label if needed
141                 OSFileStore volume = volumeMap.get(wmiVolume.getMount());
142                 volume.setName(wmiVolume.getName());
143                 if (volume.getLabel().isEmpty()) {
144                     volume.setLabel(wmiVolume.getLabel());
145                 }
146             } else if (!localOnly) {
147                 // Otherwise add the new volume in its entirety
148                 result.add(wmiVolume);
149             }
150         }
151         return result.toArray(new OSFileStore[0]);
152     }

154     /**
155      * Private method for getting all mounted local drives.
156      *
157      * @param volumeToMatch
158      *            an optional string to filter match, null otherwise
159      * @return A list of {@link OSFileStore} objects representing all local mounted
160      *         volumes
161      */
162     private static ArrayList<OSFileStore> getLocalVolumes(String volumeToMatch) {
163         ArrayList<OSFileStore> fs;
164         String volume;
165         String strFsType;
166         String strName;
167         String strMount;
168         WinNT.HANDLE hVol;
169         WinNT.LARGE_INTEGER userFreeBytes;
170         WinNT.LARGE_INTEGER totalBytes;
171         WinNT.LARGE_INTEGER systemFreeBytes;
172         boolean retVal;
173         char[] aVolume;
174         char[] fstype;
175         char[] name;
176         char[] mount;
177         IntByReference pFlags;
178 
179         fs = new ArrayList<>();
180         aVolume = new char[BUFSIZE];
181 
182         hVol = Kernel32.INSTANCE.FindFirstVolume(aVolume, BUFSIZE);
183         if (hVol == WinBase.INVALID_HANDLE_VALUE) {
184             return fs;
185         }
186 
187         while (true) {
188             fstype = new char[16];
189             name = new char[BUFSIZE];
190             mount = new char[BUFSIZE];
191             pFlags = new IntByReference();
192 
193             userFreeBytes = new WinNT.LARGE_INTEGER(0L);
194             totalBytes = new WinNT.LARGE_INTEGER(0L);
195             systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
196 
197             volume = new String(aVolume).trim();
198             Kernel32.INSTANCE.GetVolumeInformation(volume, name, BUFSIZE, null, null, pFlags, fstype, 16);
199             final int flags = pFlags.getValue();
200             Kernel32.INSTANCE.GetVolumePathNamesForVolumeName(volume, mount, BUFSIZE, null);
201 
202             strMount = new String(mount).trim();
203             if (!strMount.isEmpty() && (volumeToMatch == null || volumeToMatch.equals(volume))) {
204                 strName = new String(name).trim();
205                 strFsType = new String(fstype).trim();
206 
207                 StringBuilder options = new StringBuilder((FILE_READ_ONLY_VOLUME & flags) == 0 ? "rw" : "ro");
208                 String moreOptions = OPTIONS_MAP.entrySet().stream().filter(e -> (e.getKey() & flags) > 0)
209                         .map(Map.Entry::getValue).collect(Collectors.joining(","));
210                 if (!moreOptions.isEmpty()) {
211                     options.append(',').append(moreOptions);
212                 }
213                 Kernel32.INSTANCE.GetDiskFreeSpaceEx(volume, userFreeBytes, totalBytes, systemFreeBytes);
214                 // Parse uuid from volume name
215                 String uuid = ParseUtil.parseUuidOrDefault(volume, "");
216 
217                 // Volume is mounted
218                 OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
219                 osStore.setName(String.format("%s (%s)", strName, strMount));
220                 osStore.setVolume(volume);
221                 osStore.setLabel(strName);
222                 osStore.setMount(strMount);
223                 osStore.setDescription(getDriveType(strMount));
224                 osStore.setType(strFsType);
225                 osStore.setOptions(options.toString());
226                 osStore.setUUID(uuid);
227                 osStore.setFreeSpace(systemFreeBytes.getValue());
228                 osStore.setUsableSpace(userFreeBytes.getValue());
229                 osStore.setTotalSpace(totalBytes.getValue());
230                 fs.add(osStore);
231             }
232             retVal = Kernel32.INSTANCE.FindNextVolume(hVol, aVolume, BUFSIZE);
233             if (!retVal) {
234                 Kernel32.INSTANCE.FindVolumeClose(hVol);
235                 break;
236             }
237         }
238 
239         return fs;
240     }

242     /**
243      * Private method for getting logical drives listed in WMI.
244      *
245      * @param nameToMatch
246      *            an optional string to filter match, null otherwise
247      * @param localOnly
248      *            Whether to only search local drives
249      * @return A list of {@link OSFileStore} objects representing all network
250      *         mounted volumes
251      */
252     private static List<OSFileStore> getWmiVolumes(String nameToMatch, boolean localOnly) {
253         long free;
254         long total;
255         List<OSFileStore> fs = new ArrayList<>();
256         WmiResult<LogicalDiskProperty> drives = Win32LogicalDisk.queryLogicalDisk(nameToMatch, localOnly);
257         for (int i = 0; i < drives.getResultCount(); i++) {
258             free = WmiUtil.getUint64(drives, LogicalDiskProperty.FREESPACE, i);
259             total = WmiUtil.getUint64(drives, LogicalDiskProperty.SIZE, i);
260             String description = WmiUtil.getString(drives, LogicalDiskProperty.DESCRIPTION, i);
261             String name = WmiUtil.getString(drives, LogicalDiskProperty.NAME, i);
262             String label = WmiUtil.getString(drives, LogicalDiskProperty.VOLUMENAME, i);
263             int type = WmiUtil.getUint32(drives, LogicalDiskProperty.DRIVETYPE, i);
264             String volume;
265             if (type != 4) {
266                 char[] chrVolume = new char[BUFSIZE];
267                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(name + "\\", chrVolume, BUFSIZE);
268                 volume = new String(chrVolume).trim();
269             } else {
270                 volume = WmiUtil.getString(drives, LogicalDiskProperty.PROVIDERNAME, i);
271                 String[] split = volume.split("\\\\");
272                 if (split.length > 1 && split[split.length - 1].length() > 0) {
273                     description = split[split.length - 1];
274                 }
275             }
276             OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
277             osStore.setName(String.format("%s (%s)", description, name));
278             osStore.setVolume(volume);
279             osStore.setLabel(label);
280             osStore.setMount(name + "\\");
281             osStore.setDescription(getDriveType(name));
282             osStore.setType(WmiUtil.getString(drives, LogicalDiskProperty.FILESYSTEM, i));
283             osStore.setUUID("");
284             osStore.setFreeSpace(free); // no separate field, assume same
285             osStore.setUsableSpace(free);
286             osStore.setTotalSpace(total);
287             fs.add(osStore);
288         }
289         return fs;
290     }

292     /**
293      * Private method for getting mounted drive type.
294      *
295      * @param drive
296      *            Mounted drive
297      * @return A drive type description
298      */
299     private static String getDriveType(String drive) {
300         switch (Kernel32.INSTANCE.GetDriveType(drive)) {
301         case 2:
302             return "Removable drive";
303         case 3:
304             return "Fixed drive";
305         case 4:
306             return "Network drive";
307         case 5:
308             return "CD-ROM";
309         case 6:
310             return "RAM drive";
311         default:
312             return "Unknown drive type";
313         }
314     }

316     @Override
317     public long getOpenFileDescriptors() {
318         Map<HandleCountProperty, List<Long>> valueListMap = ProcessInformation.queryHandles().getB();
319         List<Long> valueList = valueListMap.get(HandleCountProperty.HANDLECOUNT);
320         long descriptors = 0L;
321         if (valueList != null) {
322             for (int i = 0; i < valueList.size(); i++) {
323                 descriptors += valueList.get(i);
324             }
325         }
326         return descriptors;
327     }

334     /**
335      * <p>
336      * updateFileStoreStats.
337      * </p>
338      *
339      * @param osFileStore
340      *            a {@link oshi.software.os.OSFileStore} object.
341      * @return a boolean.
342      */
343     public static boolean updateFileStoreStats(OSFileStore osFileStore) {
344         // Check if we have the volume locally
345         List<OSFileStore> volumes = getLocalVolumes(osFileStore.getVolume());
346         if (volumes.isEmpty()) {
347             // Not locally, search WMI
348             String nameToMatch = osFileStore.getMount().length() < 2 ? null : osFileStore.getMount().substring(0, 2);
349             volumes = getWmiVolumes(nameToMatch, false);
350         }
351         for (OSFileStore fileStore : volumes) {
352             if (osFileStore.getVolume().equals(fileStore.getVolume())
353                     && osFileStore.getMount().equals(fileStore.getMount())) {
354                 osFileStore.setLogicalVolume(fileStore.getLogicalVolume());
355                 osFileStore.setDescription(fileStore.getDescription());
356                 osFileStore.setType(fileStore.getType());
357                 osFileStore.setFreeSpace(fileStore.getFreeSpace());
358                 osFileStore.setUsableSpace(fileStore.getUsableSpace());
359                 osFileStore.setTotalSpace(fileStore.getTotalSpace());
360                 osFileStore.setFreeInodes(fileStore.getFreeInodes());
361                 osFileStore.setTotalInodes(fileStore.getTotalInodes());
362                 return true;
363             }
364         }
365         return false;
366     }
