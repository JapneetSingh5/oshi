129     @Override
130     public FamilyVersionInfo queryFamilyVersionInfo() {
131         WmiResult<OSVersionProperty> versionInfo = Win32OperatingSystem.queryOsVersion();
132         if (versionInfo.getResultCount() < 1) {
133             return new FamilyVersionInfo("Windows", new OSVersionInfo(System.getProperty("os.version"), null, null));
134         }
135         // Guaranteed that versionInfo is not null and lists non-empty
136         // before calling the parse*() methods
137         int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
138         String buildNumber = WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0);
139         String version = parseVersion(versionInfo, suiteMask, buildNumber);
140         String codeName = parseCodeName(suiteMask);
141         return new FamilyVersionInfo("Windows", new OSVersionInfo(version, codeName, buildNumber));
142     }

144     private static String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask, String buildNumber) {
145         // Initialize a default, sane value
146         String version = System.getProperty("os.version");
147 
148         // Version is major.minor.build. Parse the version string for
149         // major/minor and get the build number separately
150         String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.VERSION, 0).split("\\D");
151         int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
152         int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
153 
154         // see
155         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
156         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.PRODUCTTYPE,
157                 0) == WinNT.VER_NT_WORKSTATION;
158         switch (major) {
159         case 10:
160             if (minor == 0) {
161                 if (ntWorkstation) {
162                     version = "10";
163                 } else {
164                     // Build numbers greater than 17762 is Server 2019 for OS
165                     // Version 10.0
166                     version = (ParseUtil.parseLongOrDefault(buildNumber, 0L) > 17762) ? "Server 2019" : "Server 2016";
167                 }
168             }
169             break;
170         case 6:
171             if (minor == 3) {
172                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
173             } else if (minor == 2) {
174                 version = ntWorkstation ? "8" : "Server 2012";
175             } else if (minor == 1) {
176                 version = ntWorkstation ? "7" : "Server 2008 R2";
177             } else if (minor == 0) {
178                 version = ntWorkstation ? "Vista" : "Server 2008";
179             }
180             break;
181         case 5:
182             if (minor == 2) {
183                 if ((suiteMask & 0x00008000) != 0) { // VER_SUITE_WH_SERVER
184                     version = "Home Server";
185                 } else if (ntWorkstation) {
186                     version = "XP"; // 64 bits
187                 } else {
188                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
189                             : "Server 2003 R2";
190                 }
191             } else if (minor == 1) {
192                 version = "XP"; // 32 bits
193             } else if (minor == 0) {
194                 version = "2000";
195             }
196             break;
197         default:
198             break;
199         }
200 
201         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
202         if (!sp.isEmpty() && !"unknown".equals(sp)) {
203             version = version + " " + sp.replace("Service Pack ", "SP");
204         }
205 
206         return version;
207     }

209     /**
210      * Gets suites available on the system and return as a codename
211      *
212      * @param suiteMask
213      *            The suite mask bitmask
214      *
215      * @return Suites
216      */
217     private static String parseCodeName(int suiteMask) {
218         List<String> suites = new ArrayList<>();
219         if ((suiteMask & 0x00000002) != 0) {
220             suites.add("Enterprise");
221         }
222         if ((suiteMask & 0x00000004) != 0) {
223             suites.add("BackOffice");
224         }
225         if ((suiteMask & 0x00000008) != 0) {
226             suites.add("Communication Server");
227         }
228         if ((suiteMask & 0x00000080) != 0) {
229             suites.add("Datacenter");
230         }
231         if ((suiteMask & 0x00000200) != 0) {
232             suites.add("Home");
233         }
234         if ((suiteMask & 0x00000400) != 0) {
235             suites.add("Web Server");
236         }
237         if ((suiteMask & 0x00002000) != 0) {
238             suites.add("Storage Server");
239         }
240         if ((suiteMask & 0x00004000) != 0) {
241             suites.add("Compute Cluster");
242         }
243         // 0x8000, Home Server, is included in main version name
244         return String.join(",", suites);
245     }

247     @Override
248     protected int queryBitness(int jvmBitness) {
249         if (jvmBitness < 64 && System.getenv("ProgramFiles(x86)") != null && IS_VISTA_OR_GREATER) {
250             WmiResult<BitnessProperty> bitnessMap = Win32Processor.queryBitness();
251             if (bitnessMap.getResultCount() > 0) {
252                 return WmiUtil.getUint16(bitnessMap, BitnessProperty.ADDRESSWIDTH, 0);
253             }
254         }
255         return jvmBitness;
256     }

298     @Override
299     public List<OSProcess> getChildProcesses(int parentPid, int limit, ProcessSort sort) {
300         Set<Integer> childPids = new HashSet<>();
301         // Get processes from ToolHelp API for parent PID
302         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
303         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
304         try {
305             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
306                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
307                     childPids.add(processEntry.th32ProcessID.intValue());
308                 }
309             }
310         } finally {
311             Kernel32.INSTANCE.CloseHandle(snapshot);
312         }
313         List<OSProcess> procList = getProcesses(childPids);
314         List<OSProcess> sorted = processSort(procList, limit, sort);
315         return Collections.unmodifiableList(sorted);
316     }

324     private List<OSProcess> processMapToList(Collection<Integer> pids) {
325         // Get data from the registry if possible
326         Map<Integer, PerfCounterBlock> processMap = processMapFromRegistry.get();
327         // otherwise performance counters with WMI backup
328         if (processMap == null) {
329             processMap = (pids == null) ? processMapFromPerfCounters.get()
330                     : ProcessPerformanceData.buildProcessMapFromPerfCounters(pids);
331         }
332         int myPid = getProcessId();
333 
334         Map<Integer, WtsInfo> processWtsMap = ProcessWtsData.queryProcessWtsMap(pids);
335 
336         Set<Integer> mapKeys = new HashSet<>(processWtsMap.keySet());
337         mapKeys.retainAll(processMap.keySet());
338 
339         List<OSProcess> processList = new ArrayList<>();
340         for (Integer pid : mapKeys) {
341             processList.add(new WindowsOSProcess(pid, myPid, getBitness(), processMap, processWtsMap));
342         }
343         return processList;
344     }

384     private static long querySystemUptime() {
385         // Uptime is in seconds so divide milliseconds
386         // GetTickCount64 requires Vista (6.0) or later
387         if (IS_VISTA_OR_GREATER) {
388             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
389         } else {
390             // 32 bit rolls over at ~ 49 days
391             return Kernel32.INSTANCE.GetTickCount() / 1000L;
392         }
393     }

400     private static long querySystemBootTime() {
401         String eventLog = systemLog.get();
402         if (eventLog != null) {
403             try {
404                 EventLogIterator iter = new EventLogIterator(null, eventLog, WinNT.EVENTLOG_BACKWARDS_READ);
405                 // Get the most recent boot event (ID 12) from the Event log. If Windows "Fast
406                 // Startup" is enabled we may not see event 12, so also check for most recent ID
407                 // 6005 (Event log startup) as a reasonably close backup.
408                 long event6005Time = 0L;
409                 while (iter.hasNext()) {
410                     EventLogRecord record = iter.next();
411                     if (record.getStatusCode() == 12) {
412                         // Event 12 is system boot. We want this value unless we find two 6005 events
413                         // first (may occur with Fast Boot)
414                         return record.getRecord().TimeGenerated.longValue();
415                     } else if (record.getStatusCode() == 6005) {
416                         // If we already found one, this means we've found a second one without finding
417                         // an event 12. Return the latest one.
418                         if (event6005Time > 0) {
419                             return event6005Time;
420                         }
421                         // First 6005; tentatively assign
422                         event6005Time = record.getRecord().TimeGenerated.longValue();
423                     }
424                 }
425                 // Only one 6005 found, return
426                 if (event6005Time > 0) {
427                     return event6005Time;
428                 }
429             } catch (Win32Exception e) {
430                 LOG.warn("Can't open event log \"{}\".", eventLog);
431             }
432         }
433         // If we get this far, event log reading has failed, either from no log or no
434         // startup times. Subtract up time from current time as a reasonable proxy.
435         return System.currentTimeMillis() / 1000L - querySystemUptime();
436     }

443     /**
444      * Enables debug privileges for this process, required for OpenProcess() to get
445      * processes other than the current user
446      */
447     private static void enableDebugPrivilege() {
448         HANDLEByReference hToken = new HANDLEByReference();
449         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
450                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
451         if (!success) {
452             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
453             return;
454         }
455         WinNT.LUID luid = new WinNT.LUID();
456         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
457         if (!success) {
458             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
459             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
460             return;
461         }
462         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
463         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
464         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
465         if (!success) {
466             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
467         }
468         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
469     }

471     @Override
472     public OSService[] getServices() {
473         try (W32ServiceManager sm = new W32ServiceManager()) {
474             sm.open(Winsvc.SC_MANAGER_ENUMERATE_SERVICE);
475             Winsvc.ENUM_SERVICE_STATUS_PROCESS[] services = sm.enumServicesStatusExProcess(WinNT.SERVICE_WIN32,
476                     Winsvc.SERVICE_STATE_ALL, null);
477             OSServiceervice">OSService[] svcArray = new OSService[services.length];
478             for (int i = 0; i < services.length; i++) {
479                 State state;
480                 switch (services[i].ServiceStatusProcess.dwCurrentState) {
481                 case 1:
482                     state = STOPPED;
483                     break;
484                 case 4:
485                     state = RUNNING;
486                     break;
487                 default:
488                     state = OTHER;
489                     break;
490                 }
491                 svcArray[i] = new OSService(services[i].lpDisplayName, services[i].ServiceStatusProcess.dwProcessId,
492                         state);
493             }
494             return svcArray;
495         } catch (com.sun.jna.platform.win32.Win32Exception ex) {
496             LOG.error("Win32Exception: {}", ex.getMessage());
497             return new OSService[0];
498         }
499     }

501     private static String querySystemLog() {
502         String systemLog = GlobalConfig.get("oshi.os.windows.eventlog", "System");
503         if (systemLog.isEmpty()) {
504             // Use faster boot time approximation
505             return null;
506         }
507         // Check whether it works
508         HANDLE h = Advapi32.INSTANCE.OpenEventLog(null, systemLog);
509         if (h == null) {
510             LOG.warn("Unable to open configured system Event log \"{}\". Calculating boot time from uptime.",
511                     systemLog);
512             return null;
513         }
514         return systemLog;
515     }
