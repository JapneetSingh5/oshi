24  package oshi.software.common;
25  
26  import java.util.Arrays;
27  import java.util.List;
28  
29  import oshi.annotation.concurrent.ThreadSafe;
30  import oshi.software.os.FileSystem;
31  import oshi.software.os.OSFileStore;
32  
33  @ThreadSafe
34  public abstract class AbstractFileSystem implements FileSystem {
35  
36      /**
37       * FileSystem types which are network-based and should be excluded from
38       * local-only lists
39       */
40      protected static final List<String> NETWORK_FS_TYPES = Arrays.asList("afs", "cifs", "smbfs", "sshfs", "ncpfs",
41              "ncp", "nfs", "nfs4", "gfs", "gds2", "glusterfs");
42  
43      protected static final List<String> PSEUDO_FS_TYPES = Arrays.asList(//
44              // Linux defines a set of virtual file systems
45              "anon_inodefs", // anonymous inodes - inodes without filenames
46              "autofs", // automounter file system, used by Linux, Solaris, FreeBSD
47              "bdev", // keep track of block_device vs major/minor mapping
48              "binfmt_misc", // Binary format support file system
49              "bpf", // Virtual filesystem for Berkeley Paket Filter
50              "cgroup", // Cgroup file system
51              "cgroup2", // Cgroup file system
52              "configfs", // Config file system
53              "cpuset", // pseudo-filesystem interface to the kernel cpuset mechanism
54              "dax", // Direct Access (DAX) can be used on memory-backed block devices
55              "debugfs", // Debug file system
56              "devpts", // Dev pseudo terminal devices file system
57              "devtmpfs", // Dev temporary file system
58              "drm", // Direct Rendering Manager
59              "ecryptfs", // POSIX-compliant enterprise cryptographic filesystem for Linux
60              "efivarfs", // (U)EFI variable filesystem
61              "fuse", //
62              // NOTE: FUSE's fuseblk is not evalued because used as file system
63              // representation of a FUSE block storage
64              // "fuseblk" // FUSE block file system
65              "fusectl", // FUSE control file system
66              "hugetlbfs", // Huge pages support file system
67              "inotifyfs", // support inotify
68              "mqueue", // Message queue file system
69              "nfsd", // NFS file system
70              "overlay", // Overlay file system https://wiki.archlinux.org/index.php/Overlay_filesystem
71              // "pipefs", // for pipes but only visible inside kernel
72              "proc", // Proc file system, used by Linux and Solaris
73              "pstore", // Pstore file system
74              // "ramfs", // Old filesystem used for RAM disks
75              "rootfs", // Minimal fs to support kernel boot
76              "rpc_pipefs", // Sun RPC file system
77              "securityfs", // Kernel security file system
78              "selinuxfs", // SELinux file system
79              "sunrpc", // Sun RPC file system
80              "sysfs", // SysFS file system
81              "systemd-1", // Systemd file system
82              // "tmpfs", // Temporary file system
83              // NOTE: tmpfs is evaluated apart, because Linux, Solaris, FreeBSD use it for
84              // RAMdisks
85              "tracefs", // thin stackable file system for capturing file system traces
86              "usbfs", // removed in linux 3.5 but still seen in some systems
87              // FreeBSD / Solaris defines a set of virtual file systems
88              "procfs", // Proc file system
89              "devfs", // Dev temporary file system
90              "ctfs", // Contract file system
91              "fdescfs", // fd
92              "objfs", // Object file system
93              "mntfs", // Mount file system
94              "sharefs", // Share file system
95              "lofs" // Library file system
96      );
97  
98      @Override
99      public List<OSFileStore> getFileStores() {
100         return getFileStores(false);
101     }
102 }
